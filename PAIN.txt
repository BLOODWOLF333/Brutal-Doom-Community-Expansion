// Pain effects by Sergeant_Mark_IV
// Graphics can be found inside Bleed.wad

#include "zcommon.acs"
#library "paineffect"

script "BDPain" (void)
{
SetHudSize(320,200,1);

	delay(1);
	setfont("PAIN1");
	hudmessage(s:"A"; HUDMSG_FADEOUT, 0, CR_UNTRANSLATED, 150.0, 100.0, 0.5);
	delay(4);
}
