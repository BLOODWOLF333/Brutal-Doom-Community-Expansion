ACTOR BootSmearerBlue
{
	MONSTER
	-SOLID
	+THRUACTORS
	-COUNTKILL
	+LOOKALLAROUND
	-FRIENDLY
	+DONTSPLASH
	Radius 0
	Height 0
	 states
	 {

		 TNT1 A 1 A_Look

		 TNT1 A 1 A_Chase("","")

		TNT1 A 0 A_GiveToTarget("BootsSmearedWithBlueBlood", 8)

	}
}

	height 2

		TNT1 A 0 A_JumpIfInventory("IsOverMagenta", 1, "Evap")
		TNT1 A 0 A_JumpIfInventory("IsOverYava", 1, "Evap")
        TNT1 A 4
		TNT1 A 0 A_SpawnItemEx ("BootSmearerBlue",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION,0)

			BPDB A 800
			TNT1 A 0 ACS_NamedExecuteAlways("BDCheckJanitor", 0, 0, 0, 0)

			Stop

			TNT1 A 0 ACS_NamedExecuteAlways("BDCheckJanitor", 0, 0, 0, 0)
			TNT1 A 0 A_JumpIfInventory("lowgraphicsmode", 1, "Death")
			Loop

		  FadeB:
			BPDB BBBB 1 A_FadeOut(0.05)
			Stop

		  SpawnCFrameOnly:
			BPDB C 800
			TNT1 A 0 ACS_NamedExecuteAlways("BDCheckJanitor", 0, 0, 0, 0)
			TNT1 A 0 A_JumpIfInventory("lowgraphicsmode", 1, "Death")
			Loop

		  FadeC:
			BPDB CCCC 1 A_FadeOut(0.05)
			Stop

		  SpawnDFrameOnly:
			BPDB D 800
			TNT1 A 0 ACS_NamedExecuteAlways("BDCheckJanitor", 0, 0, 0, 0)
			TNT1 A 0 A_JumpIfInventory("lowgraphicsmode", 1, "Death")
			Loop

		  FadeD:
			BPDB DDDD 1 A_FadeOut(0.05)
			Stop

			TNT1 A 0
			TNT1 A 0 A_SpawnItem("BlueWaterBloodSpot")
			Stop

		Splash:
			TNT1 A 0
			TNT1 A 0 A_CustomMissile ("BlueUnderblood2", 7, 0, random (0, 360), 2, random (30, 150))

			Stop

		Death:
			TNT1 A 0
			TNT1 A 0 A_NoBlocking
			Stop
	}
}

//Base flying blood actor. Leaves bloodspots on the floor
ACTOR Blue_FlyingBlood
{
	scale 0.5
	speed 4
	radius 8
	height 1
	Gravity 0.5
	Renderstyle Translucent
	Alpha 0.9
	Decal BlueBloodSplat
	Mass 1
	+MISSILE
	+CLIENTSIDEONLY
	+NOTELEPORT
	+NOBLOCKMAP
	+THRUACTORS
	+FORCEXYBILLBOARD
	+MOVEWITHSECTOR
	+DONTSPLASH
    States
    {
     Spawn:
			TNT1 A 0
			TNT1 A 0 A_JumpIf(waterlevel > 1, "Splash")
			TNT1 A 0 A_Jump(255, "Spawn1", "Spawn2", "Spawn3", "Live")
	 Spawn1:
			TNT1 A 0
			TNT1 A 0 A_SetScale(-0.4, 0.4)
			Goto live

	Spawn2:
			TNT1 A 0
			TNT1 A 0 A_SetScale(-0.45, -0.45)
			Goto live

	Spawn3:
			TNT1 A 0
			TNT1 A 0 A_SetScale(0.5, -0.5)
			Goto live

	 Live:
		BSPB ABCDEFGHI 2
		TNT1 A 0 A_RadiusGive("BlueBloodOnVisor", 64, RGF_PLAYERS, 1)
		BSPB J 100 A_JumpIf(waterlevel > 1, "Splash")
		Stop

	Splash:
			TNT1 A 0
			TNT1 A 0 A_CustomMissile ("BlueUnderblood2", 7, 0, random (0, 360), 2, random (30, 150))
			stop

     Death:
			TNT1 A 0
			TNT1 A 0 A_PlaySound("blooddrop2")
			TNT1 A 0 A_CheckFloor("DeathFloor")
			TNT1 A 0 A_RadiusGive("BlueBloodOnVisor", 64, RGF_PLAYERS, 1)
			TNT1 A 1
			Stop

		DeathFloor:
			TNT1 A 0
			TNT1 A 0 A_SpawnItemEx ("Blue_BloodSpot",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION |  SXF_TRANSFERTRANSLATION,0)
			TNT1 A 0 A_PlaySound("blooddrop2")
			TNT1 A 0 A_RadiusGive("BlueBloodOnVisor", 64, RGF_PLAYERS, 1)
		    XDT2 EFGHIJKL 2
			Stop

		NoSpawn:
			TNT1 A 0
			Stop
    }
}
