ACTOR HandGrenades : BrutalWeapon
{
	Weapon.AmmoUse1 1
	Weapon.AmmoGive1 1
	Weapon.SelectionOrder 99999
	Weapon.AmmoType1 "GrenadeAmmo"
	Obituary "%o was shreded by %k's grenade."
    AttackSound "None"
    Inventory.PickupSound "AmmoAPIN"
	Inventory.Pickupmessage "You got a Frag Grenade! (Slot 0)"
    +WEAPON.NOAUTOAIM
    +WEAPON.NOALERT
	+WEAPON.NOAUTOFIRE
	+WEAPON.EXPLOSIVE
	+WEAPON.WIMPY_WEAPON
    Scale 0.3
	Tag "Frag Grenade"
	States
	{

	Ready:
	Ready3:
	    TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("GrenadeAmmo", 1, 1)
		Goto NoGrenades

		GRHO AB 1
		Goto Steady

	Steady:
		GRHO C 1 A_WeaponReady
		TNT1 A 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
		TNT1 A 0 A_JumpIfInventory("ExecuteDownedEnemy", 1, "DoExecution")
		TNT1 A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        TNT1 A 0 A_JumpIfInventory("Taunting",1,"Taunt")
		TNT1 A 0 A_JumpIfInventory("Salute1", 1, "Salute")
		TNT1 A 0 A_JumpIfInventory("Salute2", 1, "Salute")
		RIFG A 0 A_JumpIfInventory("IsRunning",1,"CheckSprint")
		Loop

	NoGrenades:
		TNT1 A 1 A_WeaponReady
		TNT1 A 0 A_JumpIfInventory("ExecuteDownedEnemy", 1, "DoExecution")
		TNT1 A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        TNT1 A 0 A_JumpIfInventory("Taunting",1,"Taunt")
		TNT1 A 0 A_JumpIfInventory("Salute1", 1, "Salute")
		TNT1 A 0 A_JumpIfInventory("Salute2", 1, "Salute")
		RIFG A 0 A_JumpIfInventory("IsRunning",1,"CheckSprintNoAmmo")
		Goto Ready

	Deselect:
		TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("FiredGrenade", 1, "DropGrenade")
		TNT1 A 0 A_TakeInventory("Unloading", 1)
		TNT1 A 0 A_TakeInventory("Reloading", 1)
		TNT1 A 0 A_TakeInventory("StartDualWield",1)
		TNT1 A 1 A_Lower
		Wait

	DropGrenade:
		TNT1 A 0
		TNT1 A 0 A_TakeInventory("FiredGrenade", 1)
		TNT1 A 0 A_FireCustomMissile("HandGrenadeDroped", random(-2,2), 1, 0, 0, 0, 0)
		Goto Deselect

	Select:
		TNT1 A 0
		TNT1 A 0 A_Giveinventory("GoSpecial",1)
		TNT1 A 0 A_Giveinventory("SwitchtoFist",1)
		TNT1 A 0 A_GiveInventory("FistsSelected", 1)
		TNT1 A 0 A_TakeInventory("ShotgunSelected", 1)
		TNT1 A 0 A_TakeInventory("MinigunSelected", 1)
		TNT1 A 0 A_TakeInventory("SSGSelected", 1)
		TNT1 A 0 A_TakeInventory("RocketLauncherSelected", 1)
		TNT1 A 0 A_TakeInventory("GrenadeLauncherSelected", 1)
		TNT1 A 0 A_TakeInventory("PlasmaGunSelected", 1)
		TNT1 A 0 A_TakeInventory("RailGunSelected", 1)
		TNT1 A 0 A_TakeInventory("RevenantLauncherSelected", 1)
		TNT1 A 0 A_TakeInventory("SubMachineGunSelected", 1)
		TNT1 A 0 A_TakeInventory("BFG10KSelected", 1)
		TNT1 A 0 A_TakeInventory("BFGSelected", 1)
		TNT1 A 0 A_TakeInventory("FlameCannonSelected", 1)
		TNT1 A 0 A_TakeInventory("SawSelected", 1)
		TNT1 A 0 A_Takeinventory("HasBarrel",1)
		TNT1 A 0 A_TakeInventory("TossGrenade", 1)
		TNT1 A 0 A_TakeInventory("StartDualWield", 1)
		TNT1 A 0 A_SetCrosshair(41)
		GRWG A 0 A_GunFlash
		GRWG A 1 A_Raise
		Goto Ready

	Fire:
	    GRTH ABCD 1
		TNT1 A 0 A_GiveInventory("FiredGrenade", 1)
		TNT1 A 0 A_PLaySound ("GRNPIN")
		GRTH EEFG 1

	Hold:
	    TNT1 A 1
		TNT1 A 1 A_Refire
		TNT1 A 0 A_PLaySound ("GRNTOSS")
		GRTH HI 1
		TNT1 A 0 A_FireCustomMissile("HandGrenade", 0, 1, 0, 0, 0, 0)
	    TNT1 A 0 A_TakeInventory("FiredGrenade", 1)
		GRTH JKLM 1
		TNT1 A 3
		Goto Ready

	AltFire:
		TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("GrenadeAmmo", 1, 1)
		Goto NoGrenades
		TNT1 A 0 A_TakeInventory("GrenadeAmmo", 1)
	    GRTH ABCD 1
		TNT1 A 0 A_GiveInventory("FiredGrenade", 1)
		TNT1 A 0 A_PLaySound ("GRNPIN")
		GRTH EEFG 1

	AltHold:
	    TNT1 A 1
		TNT1 A 1 A_Refire
		TNT1 A 0 A_PLaySound ("GRNTOSS")
		GRTH HI 1
		TNT1 A 0 A_FireCustomMissile("HandGrenadeAlt", 0, 1, 0, 0, 0, 0)
	    TNT1 A 0 A_TakeInventory("FiredGrenade", 1)
		GRTH JKLM 1
		TNT1 A 3
		Goto Ready

 	Spawn:
		GRND H -1
		Stop

	CheckSprint:
		GRHO C 1 A_WeaponReady(WRF_NOFIRE)
		GRND F 0 A_JumpIfInventory("IsStandingStill", 1, "Steady")
		GRND F 0 A_JumpIfInventory("IsTacticalClass", 1, "StartSprint")
		Goto Steady

	StartSprint:
		GRHO A 1 A_WeaponReady(WRF_NOFIRE)
		TNT1 A 0 A_JumpIfInventory("PowerStrength", 1, 2)
		GRND F 0 A_JumpIfInventory("UsedStamina", 90, "StopSprintTired")

	Sprinting:
		PISP A 0 offset(-9,32)
		GRND F 0 offset(-9,32) A_JumpIfInventory("IsStandingStill", 1, "Steady")
		TNT1 A 0 A_JumpIfInventory("PowerStrength", 1, 2)
		GRND F 0 offset(-9,32) A_JumpIfInventory("UsedStamina", 100, "StopSprintTired")
		PLAY A 0 offset(-9,32) ACS_NamedExecuteAlways("BDSprint", 0, 0, 0, 0)//Makes player faster.
		GRHO A 1 offset(-9,34) A_SetPitch(pitch -0.5)
		GRHO A 1 offset(-6,36) A_SetPitch(pitch -0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		GRHO A 1 offset(-3,38) A_SetPitch(pitch -0.5)
		GRHO A 1 offset(0,38) A_SetPitch(pitch -0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		GRHO A 1 offset(3,36) A_SetPitch(pitch -0.5)
		GRHO A 1 offset(6,34) A_SetPitch(pitch -0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		GRHO A 1 offset(9,32) A_SetPitch(pitch -0.5)
		GRND F 0 offset(9,32) A_WeaponReady(WRF_NOBOB)
		GRND F 0 A_JumpIfInventory("Kicking",1,"DoKick")
        GRND F 0 A_JumpIfInventory("Taunting",1,"Taunt")
		GRND F 0 offset(-9,32)
		GRND F 0 offset(9,33) A_SpawnItemEx("FootStep", 0, 0, 2, 0, 0, -4)
		GRHO A 1 offset(9,34) A_SetPitch(pitch +0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		GRHO A 1 offset(6,36) A_SetPitch(pitch +0.5)
		GRHO A 1 offset(3,38) A_SetPitch(pitch +0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		GRHO A 1 offset(0,38) A_SetPitch(pitch +0.5)
		GRHO A 1 offset(-3,36) A_SetPitch(pitch +0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		GRHO A 1 offset(-6,34) A_SetPitch(pitch +0.5)
		GRHO A 1 offset(-9,32) A_SetPitch(pitch +0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		GRND F 0 offset(-9,32) A_WeaponReady(WRF_NOBOB)
		GRND F 0 offset(-9,32) A_JumpIfInventory("IsRunning", 1, "Sprinting")
		Goto StopSprint

	StopSprintTired:
		GRHO A 0
		PLAY A 0 ACS_NamedExecuteAlways("BDTacticalSpeed", 0, 0, 0, 0)//Makes player slower.
		GRND F 0 A_PlaySound("Tired", 2)
		GRHO A 4 A_WeaponReady
		GRND F 0 A_TakeInventory("UsedStamina", 1)
		GRHO A 4 A_WeaponReady
		GRND F 0 A_TakeInventory("UsedStamina", 1)
		GRHO A 4 A_WeaponReady
		GRND F 0 A_TakeInventory("UsedStamina", 1)
		GRHO A 4 A_WeaponReady
		GRND F 0 A_TakeInventory("UsedStamina", 1)
		GRHO A 4 A_WeaponReady
		GRND F 0 A_TakeInventory("UsedStamina", 1)
		GRHO A 4 A_WeaponReady
		GRND F 0 A_TakeInventory("UsedStamina", 1)
		Goto Ready
	StopSprint:
		GRHO A 1
		TNT1 A 0 A_JumpIfInventory("PowerStrength", 1, 2)
		GRND F 0 A_JumpIfInventory("UsedStamina", 90, "StopSprintTired")
		PLAY A 0 ACS_NamedExecuteAlways("BDTacticalSpeed", 0, 0, 0, 0)//Makes player slower.
		Goto Ready

	CheckSprintNoAmmo:
		TNT1 A 1 A_WeaponReady(WRF_NOFIRE)
		TNT1 A 0 A_JumpIfInventory("IsStandingStill", 1, "Ready")
		TNT1 A 0 A_JumpIfInventory("IsTacticalClass", 1, "StartSprintNoAmmo")
		Goto Ready

	StartSprintNoAmmo:
		TNT1 A 1 A_WeaponReady(WRF_NOFIRE)
		TNT1 A 0 A_JumpIfInventory("PowerStrength", 1, 2)
		TNT1 A 0 A_JumpIfInventory("UsedStamina", 90, "StopSprintTiredNoAmmo")

	SprintingNoAmmo:
		TNT1 A 0 offset(-9,32)
		TNT1 A 0 offset(-9,32) A_JumpIfInventory("IsStandingStill", 1, "Ready")
		TNT1 A 0 A_JumpIfInventory("PowerStrength", 1, 2)
		TNT1 A 0 offset(-9,32) A_JumpIfInventory("UsedStamina", 100, "StopSprintTiredNoAmmo")
		TNT1 A 0 offset(-9,32) ACS_NamedExecuteAlways("BDSprint", 0, 0, 0, 0)//Makes player faster.
		TNT1 A 1 offset(-9,34) A_SetPitch(pitch -0.5)
		TNT1 A 1 offset(-6,36) A_SetPitch(pitch -0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		TNT1 A 1 offset(-3,38) A_SetPitch(pitch -0.5)
		TNT1 A 1 offset(0,38) A_SetPitch(pitch -0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		TNT1 A 1 offset(3,36) A_SetPitch(pitch -0.5)
		TNT1 A 1 offset(6,34) A_SetPitch(pitch -0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		TNT1 A 1 offset(9,32) A_SetPitch(pitch -0.5)
		TNT1 A 0 offset(9,32) A_WeaponReady(WRF_NOBOB)
		TNT1 A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        TNT1 A 0 A_JumpIfInventory("Taunting",1,"Taunt")
		TNT1 A 0 offset(-9,32)
		TNT1 A 0 offset(9,33) A_SpawnItemEx("FootStep", 0, 0, 2, 0, 0, -4)
		TNT1 A 1 offset(9,34) A_SetPitch(pitch +0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		TNT1 A 1 offset(6,36) A_SetPitch(pitch +0.5)
		TNT1 A 1 offset(3,38) A_SetPitch(pitch +0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		TNT1 A 1 offset(0,38) A_SetPitch(pitch +0.5)
		TNT1 A 1 offset(-3,36) A_SetPitch(pitch +0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		TNT1 A 1 offset(-6,34) A_SetPitch(pitch +0.5)
		TNT1 A 1 offset(-9,32) A_SetPitch(pitch +0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		TNT1 A 0 offset(-9,32) A_WeaponReady(WRF_NOBOB)
		TNT1 A 0 offset(-9,32) A_JumpIfInventory("IsRunning", 1, "SprintingNoAmmo")
		Goto StopSprintNoAmmo

	StopSprintTiredNoAmmo:
		TNT1 A 0
		PLAY A 0 ACS_NamedExecuteAlways("BDTacticalSpeed", 0, 0, 0, 0)//Makes player slower.
		TNT1 A 0 A_PlaySound("Tired", 2)
		TNT1 A 4 A_WeaponReady
		TNT1 A 0 A_TakeInventory("UsedStamina", 1)
		TNT1 A 4 A_WeaponReady
		TNT1 A 0 A_TakeInventory("UsedStamina", 1)
		TNT1 A 4 A_WeaponReady
		TNT1 A 0 A_TakeInventory("UsedStamina", 1)
		TNT1 A 4 A_WeaponReady
		TNT1 A 0 A_TakeInventory("UsedStamina", 1)
		TNT1 A 4 A_WeaponReady
		TNT1 A 0 A_TakeInventory("UsedStamina", 1)
		TNT1 A 4 A_WeaponReady
		TNT1 A 0 A_TakeInventory("UsedStamina", 1)
		Goto Ready
	StopSprintNoAmmo:
		TNT1 A 1
		TNT1 A 0 A_JumpIfInventory("PowerStrength", 1, 2)
		TNT1 A 0 A_JumpIfInventory("UsedStamina", 90, "StopSprintTiredNoAmmo")
		TNT1 A 0 ACS_NamedExecuteAlways("BDTacticalSpeed", 0, 0, 0, 0)//Makes player slower.
		Goto Ready
	}
}


ACTOR HandGrenade
{
	Radius 4
	Height 4
	Projectile
	Speed 30
    Damage (random (2, 2))
    Gravity 0.7
	Scale 0.4
	Projectile
	+MISSILE
    -NOGRAVITY
    -BLOODSPLATTER
	-EXTREMEDEATH
	+EXPLODEONWATER
	+SKYEXPLODE
	+BOUNCEONFLOORS
	+BOUNCEONWALLS
	+BOUNCEONCEILINGS
	-NOTELEPORT
	+MTHRUSPECIES
	+THRUSPECIES
	Species "Marines"
	BounceFactor 0.5
	WallBounceFactor 0.25
    Health 5
    SeeSound "GRNBNCE"
	DeathSound "none"
	Obituary "%o ate %k grenade."
	States
	{
	Spawn:
		GRND ABCDEFGH 2
		TNT1 A 0 A_SpawnItem("GrenadeAlerter")
		TNT1 A 0 A_GiveInventory("GrenadeTime", 1)
		TNT1 A 0 A_JumpIfInventory("GrenadeTime", 3, "Explode")
		Loop
	Death:
		TNT1 A 0
		TNT1 A 0 A_AlertMonsters(160)
	    GRND H 16 A_SpawnItem("GrenadeAlerter")
		TNT1 A 0 A_GiveInventory("GrenadeTime", 1)
		TNT1 A 0 A_JumpIfInventory("GrenadeTime", 3, "Explode")
		Loop
	Explode:
		TNT1 A 0 A_NoBlocking
		TNT1 A 0 A_SpawnItem("FragGrenadeExplosion")
		TNT1 A 0
		Stop
	}
}
