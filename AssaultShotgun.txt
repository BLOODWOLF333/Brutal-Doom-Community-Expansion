		USAR A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		USAS A 0 A_PlaySound("ASG/Reload1")
		USAR KLMN 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)

        USAS A 0 A_JumpIfInventory("AmmoShell",1,1)

        TNT1 AAA 0
		USAS A 0 A_JumpIfInventory("TurboReload", 1, "TurboReload")

		USAR ABCDEFGHIJ 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)

		Goto PutMag

		2SAR A 4 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		2SAR BCDEFG 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)

		Goto InsertBullets//empty drum

		USAS A 0 A_Takeinventory("ASGUnloaded",1)

		USAS A 0 A_JumpIfInventory("AmmoShell",1,1)
		Goto Ready
        TNT1 AAA 0
		USAS A 0 A_Takeinventory("ASGUnloaded",1)

		Goto Ready
        TNT1 AAA 0
		USAS A 0 A_Takeinventory("ASGUnloaded",1)
		USAS A 0 A_Giveinventory("AssaultShotgunAmmo",1)

		USAS A 0 A_Takeinventory("Reloading",1)

		2SAR "[]" 1
		Goto Ready

		USAS A 0 A_Takeinventory("Unloading",1)

        USAS A 0 A_JumpIfInventory("AssaultShotgunAmmo",1,2)

        TNT1 AAA 0
		USAR ABCDEFGHIJ 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		USAR A 0 A_JumpIfInventory("Kicking",1,"DoKick")

		USAR O 8 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		3SAR HGFEDCBA 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)

		USAS A 0 A_JumpIfInventory("AssaultShotgunAmmo",1,1)

        TNT1 AAA 0

		USAS A 0 A_SetCrosshair(0)
        Goto ReadyNoFuckingAmmo

		TNT1 A 0
		TNT1 A 0 A_GiveInventory("AssaultShotgun2")
