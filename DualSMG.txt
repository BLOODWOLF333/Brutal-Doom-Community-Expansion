ACTOR DualSMG : BrutalWeapon

	Weapon.AmmoUse1 0
	Weapon.AmmoGive1 0
	Weapon.AmmoUse2 0

	Weapon.AmmoType2 "BDDualSMGAmmo"

    AttackSound "None"
	Weapon.SelectionOrder 1800
    Inventory.PickupSound "Rifle/Select"
	Inventory.Pickupmessage "You got dual SMG's! (Slot 2)"
    +WEAPON.NOAUTOAIM
    +FORCEXYBILLBOARD
	+WEAPON.NO_AUTO_SWITCH

	Tag "Dual SMG's"
	DropItem "SMGSpawner"
	States
	{
	KickFlash://18 tics
		SM2S A 0 A_JumpIfInventory("SMGUnloaded", 1, "KickFlashNoAmmo")

		SM2S D 15
		SM2S CBA 1
		Stop

	KickFlashNoAmmo://18 tics
		SM2S EFG 1
		SM2S H 15
		SM2S GFE 1
		Stop

	SlideFlashEnd://10 tics
		SM2S D 0 A_JumpIfInventory("SMGUnloaded", 1, "SlideFlashEndNoAmmo")
		SM2S D 10
		SM2S CBA 1
		Stop

	SlideFlashEndNoAmmo://10 tics
		SM2S H 10
		SM2S GFE 1
		Stop

	Deselect:
		SM2F A 0 A_JumpIfInventory("SMGUnloaded", 1, "DeselectNoAmmo")
        SM2S ABCD 1
		TNT1 AAAAAAAAAAAAAAAAAA 0 A_Lower
		TNT1 A 1 A_Lower
		Wait

	DeselectNoAmmo:
        SM2S EFGH 1
		TNT1 AAAAAAAAAAAAAAAAAA 0 A_Lower
		TNT1 A 1 A_Lower
		Wait

	Select:
		TNT1 A 0 A_Giveinventory("GoSpecial",1)
		TNT1 A 0 A_TakeInventory("FistsSelected", 1)
		TNT1 A 0 A_TakeInventory("ShotgunSelected", 1)
		TNT1 A 0 A_TakeInventory("MinigunSelected", 1)
		TNT1 A 0 A_TakeInventory("SSGSelected", 1)
		TNT1 A 0 A_TakeInventory("RocketLauncherSelected", 1)
		TNT1 A 0 A_TakeInventory("GrenadeLauncherSelected", 1)
		TNT1 A 0 A_TakeInventory("PlasmaGunSelected", 1)
		TNT1 A 0 A_TakeInventory("RailGunSelected", 1)
		TNT1 A 0 A_TakeInventory("RevenantLauncherSelected", 1)
		TNT1 A 0 A_GiveInventory("SubMachineGunSelected", 1)
		TNT1 A 0 A_GiveInventory("HasDualSMG",1)
		TNT1 A 0 A_TakeInventory("BFG10KSelected", 1)
		TNT1 A 0 A_TakeInventory("BFGSelected", 1)
		TNT1 A 0 A_TakeInventory("FlameCannonSelected", 1)
		TNT1 A 0 A_TakeInventory("SawSelected", 1)
		SM2F A 0 A_Takeinventory("HasBarrel",1)
		SM2F A 0 A_Takeinventory("BrutalSMG",1)
		SM2F A 0 A_Takeinventory("StartDualWield",1)
        SM2F A 0 A_GunFlash
		TNT1 A 1 A_Raise
		TNT1 AAAAAAAAAAAA 0 A_Raise
		Goto SelectAnimation

	SelectAnimation:
		TNT1 A 0
		SM2F A 0 A_JumpIfInventory("SMGUnloaded", 1, "SelectAnimationNoAmmo")
        SM2S DCBA 1 A_JumpIfInventory("GoFatality", 1, "Steady")
        SM2F A 0 A_PlaySound("Rifle/Select")
		Goto Ready

	SelectAnimationNoAmmo:
        SM2S HGFE 1
        SM2F A 0 A_PlaySound("Rifle/Select")
		Goto Ready

	Ready3:
	Ready:
		SM2F A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SM2F A 0 A_JumpIfInventory("Taunting",1,"Taunt")
		SM2F A 0 A_JumpIfInventory("Salute1", 1, "Salute")
		SM2F A 0 A_JumpIfInventory("Salute2", 1, "Salute")
		SM2F A 0 A_JumpIfInventory("Reloading",1,"Reload")
		SM2F A 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
		TNT1 A 0 A_JumpIfInventory("ExecuteDownedEnemy", 1, "DoExecution")
		SM2F A 0 A_JumpIfInventory("Unloading",1,"Unload")
		SM2F A 0 A_JumpIfInventory("DualSMGUnloaded", 1, "ReadyNoAmmo")
		SM2F A 1 A_WeaponReady(WRF_ALLOWRELOAD)
		SM2F A 0 A_JumpIfInventory("Reloading",1,"Reload")
		SM2F A 0 A_JumpIfInventory("StartDualWield",1,"DualWield")
		SM2F A 0 A_JumpIfInventory("IsRunning",1,"StartSprint")
		Loop

	ReadyNoAmmo:
		SM2F A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SM2F A 0 A_JumpIfInventory("Taunting",1,"Taunt")
		SM2F A 0 A_JumpIfInventory("Salute1", 1, "Salute")
		SM2F A 0 A_JumpIfInventory("Salute2", 1, "Salute")
		SM2F A 0 A_JumpIfInventory("Reloading",1,"Reload")
		SM2F A 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
		TNT1 A 0 A_JumpIfInventory("ExecuteDownedEnemy", 1, "DoExecution")
		SM2F A 0 A_JumpIfInventory("Unloading",1,"Unload")
		SM2S I 1 A_WeaponReady(WRF_ALLOWRELOAD)
		SM2F A 0 A_JumpIfInventory("Reloading",1,"Reload")
		SM2F A 0 A_JumpIfInventory("StartDualWield",1,"DualWield")
		SM2F A 0 A_JumpIfInventory("IsRunning",1,"StartSprint")
		Loop

	DualWield:
		SM2F A 0 A_JumpIfInventory("SMGUnloaded", 1, "DualWieldNoAmmo")
		SM2F A 1
		SM2F A 0 A_GiveInventory("BrutalSMG",1)
		SM2F A 0 A_Takeinventory("StartDualWield",1)
		SM2F A 0 A_SelectWEapon("BrutalSMG")
		Goto Ready

	DualWieldNoAmmo:
		SM2S I 1
		SM2F A 0 A_GiveInventory("BrutalSMG",1)
		SM2F A 0 A_Takeinventory("StartDualWield",1)
		SM2F A 0 A_SelectWEapon("BrutalSMG")
		Goto Ready

	StartSprint:
		SM2F A 0 A_JumpIfInventory("SMGUnloaded", 1, "StartSprintNoAmo")
		SM2S ABC 1 A_WeaponReady(WRF_ALLOWRELOAD)
		SHTN A 0 A_Takeinventory("Zoomed",1)
        SHTN A 0 A_ZoomFactor(1.0)
		SHTN A 0 A_Takeinventory("ADSmode",1)
		RIFG A 0 A_SetCrosshair(0)
		SHTN A 0 A_Giveinventory("GoSpecial",1)
		TNT1 A 0 ACS_NamedExecuteAlways("BDSprint", 0, 0, 0, 0)//Makes player faster.
		Goto Sprinting

	StartSprintNoAmo:
		SM2S EFG 1 A_WeaponReady(WRF_ALLOWRELOAD)
		SHTN A 0 A_Takeinventory("Zoomed",1)
        SHTN A 0 A_ZoomFactor(1.0)
		SHTN A 0 A_Takeinventory("ADSmode",1)
		RIFG A 0 A_SetCrosshair(0)
		SHTN A 0 A_Giveinventory("GoSpecial",1)
		TNT1 A 0 ACS_NamedExecuteAlways("BDSprint", 0, 0, 0, 0)//Makes player faster.

	Sprinting:
		SM2S D 1 offset(-9,34) A_SetPitch(pitch -0.5)
		SM2S D 1 offset(-6,36) A_SetPitch(pitch -0.5)
		SM2S D 1 offset(-3,38) A_SetPitch(pitch -0.5)
		SM2S D 1 offset(0,38) A_SetPitch(pitch -0.5)
		SM2S D 1 offset(3,36) A_SetPitch(pitch -0.5)
		SM2S D 1 offset(6,34) A_SetPitch(pitch -0.5)
		SM2S D 1 offset(9,32) A_SetPitch(pitch -0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		SMGG A 0 A_WeaponReady(WRF_NOBOB)
		SMGG A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        SMGG A 0 A_JumpIfInventory("Taunting",1,"Taunt")
        SMGG A 0 A_JumpIfInventory("Reloading",1,"Reload")

		SM2S D 1 offset(9,34) A_SetPitch(pitch +0.5)
		SM2S D 1 offset(6,36) A_SetPitch(pitch +0.5)
		SM2S D 1 offset(3,38) A_SetPitch(pitch +0.5)
		SM2S D 1 offset(0,38) A_SetPitch(pitch +0.5)
		SM2S D 1 offset(-3,36) A_SetPitch(pitch +0.5)
		SM2S D 1 offset(-6,34) A_SetPitch(pitch +0.5)
		SM2S D 1 offset(-9,32) A_SetPitch(pitch +0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		SMGG A 0 A_WeaponReady(WRF_NOBOB)
		SMGG A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        SMGG A 0 A_JumpIfInventory("Taunting",1,"Taunt")
        SMGG A 0 A_JumpIfInventory("Reloading",1,"Reload")

		PLAY A 0 ACS_NamedExecuteAlways("BDSprint", 0, 0, 0, 0)//Makes player faster.
		SMGG A 0 A_JumpIfInventory("IsRunning", 1, "Sprinting")
		Goto StopSprint

	StopSprint:
		SM2F A 0 A_JumpIfInventory("PowerStrength", 1, 2)
		SM2F A 0 A_JumpIfInventory("UsedStamina", 100, "StopSprintTired")
		SM2F A 0 A_JumpIfInventory("SMGUnloaded", 1, "StopSprintNoAmmo")
		SM2F A 1 ACS_NamedExecuteAlways("BDTacticalSpeed", 0, 0, 0, 0)//Makes player slower.
		Goto Ready

	StopSprintNoAmmo:
		SM2S I 1 ACS_NamedExecuteAlways("BDTacticalSpeed", 0, 0, 0, 0)//Makes player slower.
		Goto Ready

	StopSprintTired:
		SM2F A 0 A_JumpIfInventory("SMGUnloaded", 1, "StopSprintTiredNoAmmo")
		SM2F A 1 ACS_NamedExecuteAlways("BDTacticalSpeed", 0, 0, 0, 0)//Makes player slower.
		SM2F A 0 A_PlaySound("Tired", 2)
		Goto Ready

	StopSprintTiredNoAmmo:
		SM2S I 1 ACS_NamedExecuteAlways("BDTacticalSpeed", 0, 0, 0, 0)//Makes player slower.
		SM2F A 0 A_PlaySound("Tired", 2)
		Goto Ready

    Fire:
        TNT1 A 0
		SM2F A 0 A_JumpIfInventory("BDSMGAmmo",1,1)
		Goto FireLeft
        TNT1 AAAA 0
		TNT1 A 0 A_FireCustomMissile("DistantFireSoundPistol", random(-1,1), 0, 0, -12, 0, random(-1,1))
        SM2F A 0 A_PlaySound("SMG/Fire")
		SM2F A 0 A_SpawnItemEx("PlayerMuzzle1",32,5,28)
		SM2F B 1 BRIGHT A_FireBullets (2.2, 1.8, -1, 12, "WeakPuff", FBF_NORANDOM)
		TNT1 A 0 A_ZoomFactor(0.99)
        TNT1 A 0 A_FireCustomMissile("DecorativeTracer",random(-2,2),0,8,0,0,random(-2,2))
		SM2F A 0 A_Takeinventory("BDSMGAmmo",1)
		SM2F A 0 A_Takeinventory("BDDualSMGAmmo",1)
		SM2F A 0 A_FireCustomMissile("PistolCaseSpawn",5,0,16,-2)
		SM2F A 0 A_SetPitch(-0.2 + pitch)
		SM2F A 1
		TNT1 A 0 A_ZoomFactor(1.0)

		SM2F A 0 A_JumpIfInventory("BDDualSMGAmmo",1,1)
        Goto NoAmmo
		TNT1 A 0 A_FireCustomMissile("DistantFireSoundPistol", random(-1,1), 0, 0, -12, 0, random(-1,1))
        SM2F A 0 A_PlaySound("SMG/Fire")
		SM2F A 0 A_SpawnItemEx("PlayerMuzzle1",32,-5,28)
		SM2F C 1 BRIGHT A_FireBullets (2.2, 1.8, -1, 12, "WeakPuff", FBF_NORANDOM)
		TNT1 A 0 A_ZoomFactor(0.99)
        TNT1 A 0 A_FireCustomMissile("DecorativeTracer",random(-2,2),0,-8,0,0,random(-2,2))
		SM2F A 0 A_Takeinventory("BDDualSMGAmmo",1)
		SM2F A 0 A_FireCustomMissile("PistolCaseSpawn",5,0,-8,-2)
		SM2F A 0 A_SetPitch(-0.2 + pitch)
		SM2F D 1
		TNT1 A 0 A_ZoomFactor(1.0)
		SM2F A 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
		Goto Ready3

		SM2F A 0 A_JumpIfInventory("BDDualSMGAmmo",1,1)

		TNT1 A 0 A_FireCustomMissile("DistantFireSoundPistol", random(-1,1), 0, 0, -12, 0, random(-1,1))

		SM2F A 0 A_SpawnItemEx("PlayerMuzzle1",32,-5,28)

        TNT1 A 0 A_FireCustomMissile("DecorativeTracer",random(-2,2),0,-8,0,0,random(-2,2))
		SM2F A 0 A_Takeinventory("BDDualSMGAmmo",1)
		SM2F A 0 A_FireCustomMissile("PistolCaseSpawn",5,0,-8,-2)
		SM2F A 0 A_SetPitch(-0.2 + pitch)
		SM2F D 1

		SM2F A 0 A_JumpIfInventory("FiredPrimary",1,"Fire")

	NoAmmo:
		SM2F A 0
		SM2F A 0 A_ZoomFactor(1.0)
		SM2F A 0 A_Takeinventory("Zoomed",1)
		SM2F A 0 A_Takeinventory("ADSmode",1)
		SM2F A 0 A_JumpIfInventory("TurboReload",1,"TurboReload")
		SM2F A 0 A_PlaySound("weapons/empty", 4)

		SM2F A 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")

		SM2F A 0 A_JumpIfInventory("Salute1", 1, "Salute")
		SM2F A 0 A_JumpIfInventory("Salute2", 1, "Salute")

		SM2S A 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")

		SM2S A 0 A_JumpIfInventory("Salute1", 1, "Salute")
		SM2S A 0 A_JumpIfInventory("Salute2", 1, "Salute")
		SM2S A 0 A_JumpIfInventory("ExecuteDownedEnemy", 1, "DoExecution")
		SM2S A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SM2S I 5 A_WeaponReady(WRF_ALLOWRELOAD | WRF_NOFIRE)
		SM2S A 0 A_JumpIfInventory("Reloading",1,"Reload")
		SM2S A 0 A_JumpIfInventory("TurboReload",1,"TurboReload")
		SM2S A 0 A_JumpIfInventory("NoAutoReload", 1, 2)
		SM2S A 0 A_JumpIfInventory("Clip1",1,"Reload")
		SM2S A 0 A_JumpIfInventory("IsRunning",1,"StartSprint")

    Reload://reload mags inserted
		"####" "#" 1
		SM2F A 0 A_Takeinventory("ADSmode",1)
		SM2F A 0 A_Takeinventory("Reloading",1)
		SM2F A 0 A_Takeinventory("Zoomed",1)
        SM2F A 0 A_ZoomFactor(1.0)
		SM2F A 0 A_JumpIfInventory("BDDualSMGAmmo",80,"Ready")

        SM2F A 0 A_JumpIfInventory("Clip1",1,3)
        Goto Ready
        TNT1 AAA 0
		SM2F A 0 A_JumpIfInventory("TurboReload",1,"TurboReload")
		SMGG A 0 A_JumpIfInventory("SMGUnloaded", 1, "PutMag")
		PIST E 0 A_JumpIfInventory("BDSMGAmmo",40,"ReloadLeft")//right mag is full
        SM2S ABCD 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		3MGR BCDE 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        SMGR FG 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SMGR H 4 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SMGR I 0 A_PlaySound("SMG/Reload1")
		SMGR IJKLMNO 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SMGR P 6 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SM2F A 0 A_GiveInventory("SMGUnloaded",1)
		SMGR A 0 A_JumpIfInventory("BDSMGAmmo",1,"PutMag")//if mag is not empty then retain it
		SM2F A 0 A_FireCustomMissile("EmptyPistolClipSpawn",0,0,10,-20)

		SMGR P 6 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        SMGR QRS 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SMGR U 0 A_PlaySound("SMG/Reload2")
		SMGR TUV 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SMGR W 4 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SMGR XYZ 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SMGR "[^]" 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		3MGR DCB 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")

        4MGR BCDEFG 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		4MGR H 4 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		4MGR I 0 A_PlaySound("SMG/Reload1")
		4MGR IJKLMNO 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		4MGR P 6 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SM2F A 0 A_GiveInventory("DualSMGUnloaded",1)
		SMGR A 0 A_JumpIfInventory("BDDualSMGAmmo",2,"PutMag2")//if mag is not empty then retain it
		SM2F A 0 A_FireCustomMissile("EmptyPistolClipSpawn",225,0,-10,-20)

	PutMag2://reload left
		4MGR P 6 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        4MGR QRS 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		4MGR U 0 A_PlaySound("SMG/Reload2")
		4MGR TUV 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		4MGR W 4 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		4MGR XYZ 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		4MGR "[^]" 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")

		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SM2F A 0 A_Takeinventory("SMGUnloaded",1)
		SM2F A 0 A_Takeinventory("DualSMGUnloaded",1)
		Goto FinishReload

	TurboReload:
		SM2F A 0 A_Takeinventory("SMGUnloaded",1)
		SM2F A 0 A_Takeinventory("DualSMGUnloaded",1)
		TNT1 A 0 A_JumpIfInventory("Clip1",1,1)
		Goto Ready
        TNT1 AAA 0
        Goto InsertBullets

	FinishReload:
        SM2S DCBA 1
		SM2F A 0 A_Takeinventory("SMGUnloaded",1)
		SM2F A 0 A_Takeinventory("DualSMGUnloaded",1)
		Goto InsertBullets

	InsertBullets:
		TNT1 AAA 0
		SM2F A 0 A_JumpIfInventory("BDDualSMGAmmo",80,8)
		SM2F A 0 A_JumpIfInventory("Clip1",1,1)
		Goto Ready
        TNT1 AAA 0
		SM2F A 0 A_Giveinventory("BDSMGAmmo",1)
		SM2F A 0 A_Giveinventory("BDDualSMGAmmo",1)
		SM2F A 0 A_Takeinventory("Clip1",1)
		Goto InsertBullets

		TNT1 AAA 0
		SM2F A 1 A_Takeinventory("Reloading",1)
		SM2F A 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
        Goto Ready

 	 Unload:
		"####" "#" 1 A_WeaponReady
		SM2F A 0 A_Takeinventory("ADSmode",1)
		SM2F A 0 A_Takeinventory("Unloading",1)
		SM2F A 0 A_Takeinventory("Zoomed",1)
        PI2G A 0 A_ZoomFactor(1.0)
		PI2G A 0 A_GiveInventory ("Pumping", 1)
		SM2F A 0 A_Takeinventory("Reloading",1)
		SM2F A 0 A_JumpIfInventory("SMGUnloaded",1,"UnloadLeft")
        SM2S ABCD 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
