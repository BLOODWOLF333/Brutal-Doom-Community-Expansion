ACTOR SSG : BrutalWeapon
{
	Game Doom
	SpawnID 33
	Weapon.Kickback 50
	Weapon.SelectionOrder 400
	Weapon.AmmoGive1 0
	Weapon.AmmoGive2 0
	Weapon.AmmoType1 "AmmoShell"
	Weapon.AmmoType2 "SSGAmmo"
	Weapon.AmmoUse1 0
	Weapon.AmmoUse2 0
	+WEAPON.NOAUTOAIM
	+WEAPON.NOAUTOFIRE
	+FORCEXYBILLBOARD
	Inventory.PickupMessage "$GOTSHOTGUN2"
	Obituary "$OB_MPSSHOTGUN"
	Inventory.PickupSound "SSG/Draw"
	YScale 0.8
	XScale 1.0
	Tag "Super Shotgun"
	DropItem "SSGSpawner"
	States
	{
	KickFlash://18 tics
		SHT2 OPQ 1
		SHT2 R 15
		SHT2 QPO 1
		Stop

	SlideFlashEnd://10 tics
		SHT2 R 10
		SHT2 QPO 1
		Stop

	Ready:
	Ready3:
	    SHT2 A 1 A_WeaponReady(WRF_ALLOWRELOAD)
        SHT2 A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        SHT2 A 0 A_JumpIfInventory("Taunting",1,"Taunt")
		SHT2 A 0 A_JumpIfInventory("Salute1", 1, "Salute")
		SHT2 A 0 A_JumpIfInventory("Salute2", 1, "Salute")
		SHT2 A 0 A_JumpIfInventory("Reloading",1,"Reload")
		SHT2 A 0 A_JumpIfInventory("Unloading",1,"Unload")
		SHT2 A 0 A_JumpIfInventory("StartDualWield",1,"ChangeShell")
		SHT2 A 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
		TNT1 A 0 A_JumpIfInventory("ExecuteDownedEnemy", 1, "DoExecution")
		SHT2 A 0 A_JumpIfInventory("IsRunning",1,"StartSprint")
		Loop

	ChangeShell:
		TNT1 A 0 A_TakeInventory("StartDualWield", 1)
		TNT1 A 0 A_GiveInventory("ShotgunChangeAmmo", 1)
        TNT1 A 0 A_ZoomFactor(1.0)
		TNT1 A 0 A_JumpIfInventory("AmmoSlug", 1, "Unload")
		Goto NoSlugs

	NoSlugs:
		TNT1 A 0 A_Print("You are out of shotgun slugs.", 2)
		SHTN A 0 A_TakeInventory("ShotgunChangeAmmo",1)
		SHT2 A 3
		Goto Ready

	StartSprint:
		SHT2 STU 1 A_WeaponReady(WRF_ALLOWRELOAD)
		SHTN A 0 A_Takeinventory("Zoomed",1)
        SHTN A 0 A_ZoomFactor(1.0)
		SHTN A 0 A_Takeinventory("ADSmode",1)
		RIFG A 0 A_SetCrosshair(0)
		SHTN A 0 A_Giveinventory("GoSpecial",1)
		TNT1 A 0 ACS_NamedExecuteAlways("BDSprint", 0, 0, 0, 0)//Makes player faster.)

	Sprinting:
		SHT2 V 1 offset(-9,34) A_SetPitch(pitch -0.5)
		SHT2 V 1 offset(-6,36) A_SetPitch(pitch -0.5)
		SHT2 V 1 offset(-3,38) A_SetPitch(pitch -0.5)
		SHT2 V 1 offset(0,38) A_SetPitch(pitch -0.5)
		SHT2 V 1 offset(3,36) A_SetPitch(pitch -0.5)
		SHT2 V 1 offset(6,34) A_SetPitch(pitch -0.5)
		SHT2 V 1 offset(9,32) A_SetPitch(pitch -0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 2)
		SMGG A 0 A_WeaponReady(WRF_NOBOB)
		SMGG A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        SMGG A 0 A_JumpIfInventory("Taunting",1,"Taunt")
        SMGG A 0 A_JumpIfInventory("Reloading",1,"Reload")

		SHT2 V 1 offset(9,34) A_SetPitch(pitch +0.5)
		SHT2 V 1 offset(6,36) A_SetPitch(pitch +0.5)
		SHT2 V 1 offset(3,38) A_SetPitch(pitch +0.5)
		SHT2 V 1 offset(0,38) A_SetPitch(pitch +0.5)
		SHT2 V 1 offset(-3,36) A_SetPitch(pitch +0.5)
		SHT2 V 1 offset(-6,34) A_SetPitch(pitch +0.5)
		SHT2 V 1 offset(-9,32) A_SetPitch(pitch +0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 2)
		SMGG A 0 A_WeaponReady(WRF_NOBOB)
		SMGG A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        SMGG A 0 A_JumpIfInventory("Taunting",1,"Taunt")
        SMGG A 0 A_JumpIfInventory("Reloading",1,"Reload")

		PLAY A 0 ACS_NamedExecuteAlways("BDSprint", 0, 0, 0, 0)//Makes player faster.
		SMGG A 0 A_JumpIfInventory("IsRunning", 1, "Sprinting")
		Goto StopSprint

	StopSprint:
		SSRU A 0 A_JumpIfInventory("PowerStrength", 1, 2)
		SHT2 A 0 A_JumpIfInventory("UsedStamina", 100, "StopSprintTired")
		SHT2 UTS 1 ACS_NamedExecuteAlways("BDTacticalSpeed", 0, 0, 0, 0)//Makes player slower.
		Goto Ready

	StopSprintTired:
		SHT2 UTS 1 ACS_NamedExecuteAlways("BDTacticalSpeed", 0, 0, 0, 0)//Makes player slower.
		SHT2 A 0 A_PlaySound("Tired", 2)
		Goto Ready

	Deselect:
		SHT2 A 0 A_Takeinventory("Zoomed",1)
        SHT2 A 0 A_ZoomFactor(1.0)
		SHT2 A 0 A_TakeInventory("TossGrenade", 1)
		SHT2 OPQR 1
		TNT1 AAAAAAAAAAAAAAAAAA 0 A_Lower
		TNT1 A 1 A_Lower
		Wait

	Select:
		SHT2 A 0 A_Giveinventory("GoSpecial",1)
		TNT1 A 0 A_TakeInventory("FistsSelected", 1)
		TNT1 A 0 A_TakeInventory("ShotgunSelected", 1)
		TNT1 A 0 A_TakeInventory("MinigunSelected", 1)
		TNT1 A 0 A_GiveInventory("SSGSelected", 1)
		TNT1 A 0 A_TakeInventory("RocketLauncherSelected", 1)
		TNT1 A 0 A_TakeInventory("GrenadeLauncherSelected", 1)
		TNT1 A 0 A_TakeInventory("PlasmaGunSelected", 1)
		TNT1 A 0 A_TakeInventory("RailGunSelected", 1)
		TNT1 A 0 A_TakeInventory("RevenantLauncherSelected", 1)
		TNT1 A 0 A_TakeInventory("SubMachineGunSelected", 1)
		TNT1 A 0 A_TakeInventory("BFG10KSelected", 1)
		TNT1 A 0 A_TakeInventory("BFGSelected", 1)
		TNT1 A 0 A_TakeInventory("FlameCannonSelected", 1)
		TNT1 A 0 A_TakeInventory("SawSelected", 1)
		SHT2 A 0 A_Takeinventory("HasBarrel",1)
		SHT2 A 0 A_TakeInventory("TossGrenade", 1)
		SHT2 A 0 A_TakeInventory("SSG2", 1)
		SHT2 A 0 A_GunFlash
		TNT1 A 1 A_Raise
		TNT1 AAAAAAAAAAAA 0 A_Raise
		SHT2 A 0 A_JumpIfInventory("SSGAmmo", 1, 1)
		Goto RaiseReloading
		TTN1 AAA 0
		Goto SelectAnimation

    SelectAnimation:
		SHT2 KLMN 1 A_JumpIfInventory("GoFatality", 1, "Steady")
		SHT2 A 0 A_PlaySound("SSG/Draw")
		Goto Ready

	RaiseReloading:
		SHT2 A 0
		SHT2 A 0 A_JumpIfInventory("AmmoShell", 2, 1)
		Goto RaiseNoAmmo
		TNT1 AAA 0
		SHT2 A 0 A_playsound("weapons/sshoto")
		SHTZ LMNOPQRSTUV 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)

		SHT2 A 0 A_playsound("weapons/sshotl")
		SHTZ WXY 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		TNT1 A 4 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		SHT2 A 0 A_TakeInventory("AmmoShell", 2)
		SHT2 A 0 A_GiveInventory("SSGAmmo", 2)

		SHTZ Z 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		SHT2 A 0 A_PlaySound("weapons/sshotc")
		SHZ2 AB 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		SHZ2 C 4 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		SHZ2 DEFGHIJ 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)

		SHT2 A 0 A_TakeInventory("Pumping", 1)
		SHT2 A 0 A_TakeInventory("SSGUnloaded", 1)
		SHT2 A 0 A_TakeInventory("SSGAlt",1)
		TNT1 A 0 A_TakeInventory("ShotgunChangeAmmo",1)
		SHTZ BA 1 A_ReFire
		Goto Ready

	RaiseNoAmmo:
		SHT2 KLMN 1
		SHT2 A 0 A_PlaySound("weapons/sshotc")
		Goto Ready

	NoAmmo:
		SHT2 A 0
		SHT2 A 0 A_ZoomFactor(1.0)
		SHT2 A 0 A_Takeinventory("Zoomed",1)
		SHT2 A 0 A_Takeinventory("ADSmode",1)
		TNT1 A 0 A_PlaySound("weapons/empty", 4)
		NoAmmo2:
		SHT2 A 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
	    SHT2 A 0 A_JumpIfInventory("Taunting",1,"Taunt")
		SHT2 A 0 A_JumpIfInventory("Salute1", 1, "Salute")
		SHT2 A 0 A_JumpIfInventory("Salute2", 1, "Salute")
		SHT2 A 0 A_JumpIfInventory("ExecuteDownedEnemy", 1, "DoExecution")
		SHT2 A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHT2 A 5 A_WeaponReady(WRF_ALLOWRELOAD | WRF_NOFIRE)
		SHT2 A 0 A_JumpIfInventory("Reloading",1,"Reload")
		TNT1 A 0 A_JumpIfInventory("NoAutoReload", 1, 2)
		SHT2 A 0 A_JumpIfInventory("AmmoShell",1,"Reload")
		SHT2 A 0 A_JumpIfInventory("IsRunning",1,"StartSprint")
		Goto Ready3

	 Fire:
		SHT2 A 0 A_TakeInventory("Reloading",1)
		SHT2 A 0 A_JumpIfInventory("SSGAmmo", 2, 2)
		SHT2 A 0 A_JumpIfInventory("SSGAmmo", 1, "AltFire")
		Goto NoAmmo
		TNT1 AAA 0
		SHT2 A 0 A_TakeInventory("SSGAmmo", 2)
		SHT2 A 0 A_JumpIfInventory("PowerStrength", 1, 2)
		SHT2 A 0 A_Recoil(3)
		SHT2 A 0
		SHT2 A 0 A_FireCustomMissile("DistantFireSoundSSG", random(-1,1), 0, 0, -12, 0, random(-1,1))
		SHO9 A 0 A_firebullets (0,0,1,40,"shotpuff",FBF_NORANDOM,128)
        SHO9 A 0 A_playsound("SSG/Fire", 1)
		TNT1 AAAAAAAAAAA 0 A_FireCustomMissile("DecorativeTracer", random(-6,6), 0, 0, 0, 0, random(-3,3))
		TNT1 AAAAAAAAAAAAAAAAAAAA 0 A_FireCustomMissile("ShotgunParticles", random(-18,18), 0, -1, 0, 0, random(-9,9))
		TNT1 AAAAAAAAAAAAA 0 A_FireCustomMissile("ShotgunParticles2", random(-18,18), 0, -1, 0, 0, random(-9,9))
		SHT2 A 0 A_SpawnItemEx("PlayerMuzzle1",32,0,28)
		TNT1 A 0 A_FireCustomMissile("Alerter",0,0)
		SHT2 B 1 BRIGHT
		SHT2 A 0 A_FireBullets (5.6, 2.8, 9, 13, "ShotgunPuff", FBF_NORANDOM)
		SHT2 A 0 A_FireBullets (5.6, 2.8, 9, 13, "ShotgunPuff", FBF_NORANDOM)
		TNT1 A 0 A_ZoomFactor(0.96)
		SHT2 A 0 A_JumpIfInventory("IsNOTTacticalClass", 1, 3)//Tactical Mode deals extra damage and recoil
		SHTN A 0 A_FireBullets (5.6, 2.8, 10, 13, "ShotgunPuff2", FBF_NORANDOM)
		SHT2 A 0 A_SetPitch(pitch-3.6)
		SHT2 A 0 A_SetPitch(pitch-3.6)
        SHT2 C 1 BRIGHT
		TNT1 A 0 A_ZoomFactor(1.0)
        SHT2 A 0 A_TakeInventory("SSGAlt",1)
        SHT2 A 0 A_TakeInventory("Reloading",1)

		SHT2 DEFGHI 1 A_SetPitch(pitch+0.6)

		SHT2 A 0 A_JumpIfInventory("Reloading",1,"Reload")
		SHT2 A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        SHT2 A 0 A_JumpIfInventory("Taunting",1,"Taunt")
		SHT2 A 0 A_JumpIfInventory("Salute1", 1, "Salute")
		SHT2 A 0 A_JumpIfInventory("Salute2", 1, "Salute")
		SHT2 A 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
		SHT2 A 1 A_WeaponReady(WRF_NOFIRE | WRF_NOBOB)
		TNT1 A 0 A_JumpIfInventory("NoAutoReload", 1, 1)//Don't go to NoAmmo unless they try to fire again
		Goto NoAmmo2
		TNT1 AAA 0
		Goto Ready

	Reload:	//both barrels
		SHT2 A 0
		SHT2 A 0 A_TakeInventory("Reloading",1)
		SHT2 A 0 A_JumpIfInventory("AmmoShell", 1, 1)//Check if has any shells in inventory
		Goto Ready
		TNT1 AAA 0
		SHT2 A 0 A_JumpIfInventory("AmmoShell", 2, 1)//Check if has two shells to reload both barrels
		Goto Reload2
		TNT1 AAA 0
		SHT2 A 0 A_JumpIfInventory("SSGAlt", 1, "Reload2")//Check if right shell has been fired
		SHT2 A 0 A_JumpIfInventory("SSGAmmo", 2, "Ready")//Check if both barrels are loaded
		SHT2 A 0 A_JumpIfInventory("SSGAmmo", 1, "Reload2")

		SHTZ ABCDEFGH 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHTZ I 4 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHTZ J 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHT2 A 0 A_GiveInventory("Pumping", 1)

		SHT2 A 0 A_PlaySound("weapons/sshoto")
		SHT2 A 0 A_JumpIfInventory("SSGUnloaded", 1, 4)
		SHTZ K 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		TNT1 AA 0 A_FireCustomMissile("SSGCaseSpawner",0,0,-8, -8)
		SHTZ L 4 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")

		SHTZ MNOPQRSTUV 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")

		SHTZ WXY 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		TNT1 A 4 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHT2 A 0 A_TakeInventory("AmmoShell", 2)
		SHT2 A 0 A_GiveInventory("SSGAmmo", 2)

		SHTZ Z 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHT2 A 0 A_PlaySound("weapons/sshotc")
		SHZ2 AB 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHZ2 C 4 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHZ2 DEFGHIJ 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")

		SHT2 A 0 A_TakeInventory("Pumping", 1)
		SHT2 A 0 A_TakeInventory("SSGUnloaded", 1)
		SHT2 A 0 A_TakeInventory("SSGAlt",1)

        TNT1 AAA 0
		SHTZ ABCDEFGH 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHTZ I 4 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHTZ J 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")

		SHT2 A 0 A_PlaySound("weapons/sshoto")
		SHTZ K 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHZ2 A 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")

		SHTZ X 4 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHTZ W 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHT2 A 0 A_PlaySound("weapons/sshotl")
		SHTZ VUTSRQPONM 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHTZ L 4 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")

	RemoveBullets:
		TNT1 AAAA 0
		SHT2 A 0 A_JumpIfInventory("SSGAmmo",1,1)
		Goto FinishUnload
        TNT1 AAA 0
		SHT2 A 0 A_Takeinventory("SSGAmmo",1)
		SHT2 A 0 A_Giveinventory("AmmoShell",1)
		Goto RemoveBullets

	FInishUnload:
		SHT2 A 0 A_PlaySound("weapons/sshotc")
		SHTZ J 1
		SHTZ I 4
		SHTZ HGFEDCBA 1
		SHTZ A 0 A_GiveInventory("SSGUnloaded", 1)
		SHTZ A 0 A_Takeinventory("Unloading",1)
		SHTZ A 0 A_TakeInventory("SSGAlt",1)
		TNT1 A 0 A_JumpIfInventory("ShotgunChangeAmmo", 1, "ChangeAmmo")
		Goto Ready

	ChangeAmmo:
		TNT1 A 0
		TNT1 A 0 A_GiveInventory("SSG2")
		TNT1 A 0 A_GiveInventory("Reloading",1)
		TNT1 A 0 A_SelectWeapon("SSG2")
		Goto Ready3
	}
