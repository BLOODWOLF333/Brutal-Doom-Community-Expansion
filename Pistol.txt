ACTOR BrutalPistol : BrutalWeapon
{
	Weapon.AmmoUse1 0
	Weapon.AmmoGive1 0
	Weapon.AmmoUse2 0
	Weapon.AmmoGive2 0
	Weapon.AmmoType1 "Clip1"
	Weapon.AmmoType2 "BDPistolAmmo"
	weapon.selectionorder 1900
	Obituary "%o was shot down by %k's pistol."
    AttackSound "None"
    Inventory.PickupSound "Pistol/Draw"
	Inventory.Pickupmessage "$GOTPISTOL"
    +WEAPON.NOAUTOAIM
	+WEAPON.NOAUTOFIRE
	+WEAPON.WIMPY_WEAPON
    +FORCEXYBILLBOARD
	+WEAPON.NOALERT
	+WEAPON.NO_AUTO_SWITCH
	Scale 0.9
	Tag "Pistol"
	DropItem "PistolSpawner"
	States
	{
	Ready3:
	Ready:
		PIST E 0 A_JumpIfInventory("BDPistolAmmo",1,1)
		Goto ReadyNoAmmo
		TNT1 AAA 0
		PIST E 1 A_WeaponReady(WRF_ALLOWRELOAD)
        PIST E 0 A_JumpIfInventory("Kicking",1,"DoKick")
        PIST E 0 A_JumpIfInventory("Taunting",1,"Taunt")
		PIST E 0 A_JumpIfInventory("Salute1", 1, "Salute")
		PIST E 0 A_JumpIfInventory("Salute2", 1, "Salute")
        PIST E 0 A_JumpIfInventory("Reloading",1,"Reload")
		PIST E 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
		TNT1 A 0 A_JumpIfInventory("ExecuteDownedEnemy", 1, "DoExecution")
		PIST E 0 A_JumpIfInventory("Unloading",1,"Unload")
		PIST E 0 A_Takeinventory("PistolFireAnimation1",1)
		PIST E 0 A_JumpIfInventory("StartDualWield",1,"DualWield")
		PIST E 0 A_JumpIfInventory("IsRunning",1,"StartSprint")
		Loop

	ReadyNoAmmo:
		PIST I 1 A_WeaponReady(WRF_ALLOWRELOAD)
        PIST E 0 A_JumpIfInventory("Kicking",1,"DoKick")
        PIST E 0 A_JumpIfInventory("Taunting",1,"Taunt")
		PIST E 0 A_JumpIfInventory("Salute1", 1, "Salute")
		PIST E 0 A_JumpIfInventory("Salute2", 1, "Salute")
        PIST E 0 A_JumpIfInventory("Reloading",1,"Reload")
		PIST E 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
		TNT1 A 0 A_JumpIfInventory("ExecuteDownedEnemy", 1, "DoExecution")
		PIST E 0 A_JumpIfInventory("Unloading",1,"Unload")
		PIST E 0 A_Takeinventory("PistolFireAnimation1",1)
		PIST E 0 A_JumpIfInventory("StartDualWield",1,"DualWield")
		PIST E 0 A_JumpIfInventory("IsRunning",1,"StartSprint")//Make a SprintNoAmmo state(s)
		Loop

	StartSprint:
		PISP ABC 1 A_WeaponReady(WRF_ALLOWRELOAD)
		SHTN A 0 A_Takeinventory("Zoomed",1)
        SHTN A 0 A_ZoomFactor(1.0)
		SHTN A 0 A_Takeinventory("ADSmode",1)
		RIFG A 0 A_SetCrosshair(0)
		SHTN A 0 A_Giveinventory("GoSpecial",1)
		TNT1 A 0 ACS_NamedExecuteAlways("BDSprint", 0, 0, 0, 0)//Makes player faster.

	Sprinting:
		PISP D 1 offset(-9,34) A_SetPitch(pitch -0.5)
		PISP D 1 offset(-6,36) A_SetPitch(pitch -0.5)
		PISP D 1 offset(-3,38) A_SetPitch(pitch -0.5)
		PISP D 1 offset(0,38) A_SetPitch(pitch -0.5)
		PISP D 1 offset(3,36) A_SetPitch(pitch -0.5)
		PISP D 1 offset(6,34) A_SetPitch(pitch -0.5)
		PISP D 1 offset(9,32) A_SetPitch(pitch -0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		SMGG A 0 A_WeaponReady(WRF_NOBOB)
		SMGG A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        SMGG A 0 A_JumpIfInventory("Taunting",1,"Taunt")
        SMGG A 0 A_JumpIfInventory("Reloading",1,"Reload")

		PISP D 1 offset(9,34) A_SetPitch(pitch +0.5)
		PISP D 1 offset(6,36) A_SetPitch(pitch +0.5)
		PISP D 1 offset(3,38) A_SetPitch(pitch +0.5)
		PISP D 1 offset(0,38) A_SetPitch(pitch +0.5)
		PISP D 1 offset(-3,36) A_SetPitch(pitch +0.5)
		PISP D 1 offset(-6,34) A_SetPitch(pitch +0.5)
		PISP D 1 offset(-9,32) A_SetPitch(pitch +0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 1)
		SMGG A 0 A_WeaponReady(WRF_NOBOB)
		SMGG A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        SMGG A 0 A_JumpIfInventory("Taunting",1,"Taunt")
        SMGG A 0 A_JumpIfInventory("Reloading",1,"Reload")

		PLAY A 0 ACS_NamedExecuteAlways("BDSprint", 0, 0, 0, 0)//Makes player faster.
		SMGG A 0 A_JumpIfInventory("IsRunning", 1, "Sprinting")
		Goto StopSprint

	StopSprint:
		PISP D 0 A_JumpIfInventory("PowerStrength", 1, 2)
		PIST E 0 A_JumpIfInventory("UsedStamina", 100, "StopSprintTired")
		PISP CB 1
		PISP A 1 ACS_NamedExecuteAlways("BDTacticalSpeed", 0, 0, 0, 0)//Makes player slower.
		Goto Ready

	StopSprintTired:
		PISP CB 1
		PISP A 1 ACS_NamedExecuteAlways("BDTacticalSpeed", 0, 0, 0, 0)//Makes player slower.
		PIST E 0 A_PlaySound("Tired", 2)
		Goto Ready

	Deselect:
        PISS CBA 1
		TNT1 AAAAAAAAAAAAAAAAAA 0 A_Lower
		PIST E 0 A_TakeInventory("TossGrenade", 1)
		TNT1 A 1 A_Lower
		Wait

	Select:
		PIST E 0 A_Takeinventory("StartDualWield",1)
		PIST E 0 A_Giveinventory("GoSpecial",1)
		TNT1 A 0 A_TakeInventory("FistsSelected", 1)
		TNT1 A 0 A_TakeInventory("ShotgunSelected", 1)
		TNT1 A 0 A_TakeInventory("MinigunSelected", 1)
		TNT1 A 0 A_TakeInventory("SSGSelected", 1)
		TNT1 A 0 A_TakeInventory("RocketLauncherSelected", 1)
		TNT1 A 0 A_TakeInventory("GrenadeLauncherSelected", 1)
		TNT1 A 0 A_TakeInventory("PlasmaGunSelected", 1)
		TNT1 A 0 A_TakeInventory("RailGunSelected", 1)
		TNT1 A 0 A_TakeInventory("RevenantLauncherSelected", 1)
		TNT1 A 0 A_GiveInventory("SubMachineGunSelected", 1)
		TNT1 A 0 A_TakeInventory("BFG10KSelected", 1)
		TNT1 A 0 A_TakeInventory("BFGSelected", 1)
		TNT1 A 0 A_TakeInventory("FlameCannonSelected", 1)
		TNT1 A 0 A_TakeInventory("SawSelected", 1)
		PIST E 0 A_Takeinventory("HasBarrel",1)
		PIST E 0 A_Takeinventory("DualPistols",1)
        PIST E 0 A_GunFlash
		TNT1 A 1 A_Raise
		TNT1 AAAAAAAAAA 0 A_Raise
		Goto SelectAnimation

	SelectAnimation:
		TNT1 A 1
	    PIST E 0 A_JumpIfInventory("GoFatality", 1, "Steady")
        PIST E 0 A_PlaySound("Pistol/Draw")
        PISS ABC 1
		PIST E 0 A_Takeinventory("StartDualWield",1)
		Goto Ready

    Fire:
        PIST E 0 A_JumpIfInventory("BDPistolAmmo",1,2)
        Goto NoAmmo
        TNT1 AAAA 0
		PIST E 0 A_AlertMonsters

		TNT1 A 0 A_JumpIFInventory("PistolBurstFire", 1, "Fire2")
		TNT1 A 0 A_FireCustomMissile("DistantFireSoundPistol", random(-1,1), 0, 0, -12, 0, random(-1,1))
        PIST E 0 A_PlaySound("Pistol/Fire", 1)
		PIST E 0 A_SpawnItemEx("PlayerMuzzle1",32,0,28)
		PIST A 1 BRIGHT A_FireBullets (0.8, 0.9, -1, 14, "WeakPuff", FBF_NORANDOM)
		TNT1 A 0 A_ZoomFactor(0.99)
        TNT1 A 0 A_FireCustomMissile("DecorativeTracer",random(-1,1),0,0,0,0,random(-1,1))
		PIST E 0 A_Takeinventory("BDPistolAmmo",1)
		PIST E 0 A_Takeinventory("BDDualPistolAmmo",1)
        PIST B 1
		TNT1 A 0 A_ZoomFactor(1.0)
		PIST E 0 A_FireCustomMissile("PistolCaseSpawn",5,0,0,-9)
		PIST C 1
		PIST E 0 A_JumpIfInventory("IsTacticalClass", 1, "FireEndTactical")
		PIST D 1
		PIST E 0 A_JumpIfInventory("Reloading",1,"Reload")
		PIST E 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
		PIST E 2 A_WeaponReady(WRF_NOBOB | WRF_ALLOWRELOAD)
		PIST E 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
		Goto Ready

	Fire2:
        TNT1 A 0
		TNT1 A 0 A_FireCustomMissile("DistantFireSoundPistol", random(-1,1), 0, 0, -12, 0, random(-1,1))
        PIST E 0 A_PlaySound("Pistol/Fire", 1)
		PIST E 0 A_SpawnItemEx("PlayerMuzzle1",32,0,28)
		PIST A 1 BRIGHT A_FireBullets (0.8, 0.9, -1, 14, "WeakPuff", FBF_NORANDOM)
		TNT1 A 0 A_ZoomFactor(0.99)
		TNT1 A 0 A_SetPitch(pitch-0.2)
        TNT1 A 0 A_FireCustomMissile("DecorativeTracer",random(-1,1),0,0,0,0,random(-1,1))
		PIST E 0 A_Takeinventory("BDPistolAmmo",1)
		PIST E 0 A_Takeinventory("BDDualPistolAmmo",1)
		PIST E 0 A_JumpIfInventory("BDPistolAmmo",1,2)
		Goto NoAmmo
        PIST C 1
		TNT1 A 0 A_ZoomFactor(1.0)
		PIST E 0 A_FireCustomMissile("PistolCaseSpawn",5,0,0,-9)
		PIST E 1
		TNT1 A 0 A_FireCustomMissile("DistantFireSoundPistol", random(-1,1), 0, 0, -12, 0, random(-1,1))
        PIST E 0 A_PlaySound("Pistol/Fire", 1)
		PIST E 0 A_SpawnItemEx("PlayerMuzzle1",32,0,28)
		PIST A 1 BRIGHT A_FireBullets (1.0, 1.1, -1, 14, "WeakPuff", FBF_NORANDOM)
		TNT1 A 0 A_ZoomFactor(0.99)
		TNT1 A 0 A_SetPitch(pitch-0.2)
        TNT1 A 0 A_FireCustomMissile("DecorativeTracer",random(-1,1),0,0,0,0,random(-1,1))
		PIST E 0 A_Takeinventory("BDPistolAmmo",1)
		PIST E 0 A_Takeinventory("BDDualPistolAmmo",1)
		PIST E 0 A_JumpIfInventory("BDPistolAmmo",1,2)
		Goto NoAmmo
        PIST B 0
        PIST C 1
		TNT1 A 0 A_ZoomFactor(1.0)
		PIST E 0 A_FireCustomMissile("PistolCaseSpawn",5,0,0,-9)
		PIST E 1
		TNT1 A 0 A_FireCustomMissile("DistantFireSoundPistol", random(-1,1), 0, 0, -12, 0, random(-1,1))
        PIST E 0 A_PlaySound("Pistol/Fire", 1)
		PIST E 0 A_SpawnItemEx("PlayerMuzzle1",32,0,28)
		PIST A 1 BRIGHT A_FireBullets (1.3, 1.4, -1, 14, "WeakPuff", FBF_NORANDOM)
		TNT1 A 0 A_ZoomFactor(0.99)
		TNT1 A 0 A_SetPitch(pitch-0.2)
        TNT1 A 0 A_FireCustomMissile("DecorativeTracer",random(-1,1),0,0,0,0,random(-1,1))
		PIST E 0 A_Takeinventory("BDPistolAmmo",1)
		PIST E 0 A_Takeinventory("BDDualPistolAmmo",1)
		PIST E 0 A_JumpIfInventory("BDPistolAmmo",1,2)
		Goto NoAmmo
        PIST B 0
        PIST C 1
		TNT1 A 0 A_ZoomFactor(1.0)
		PIST E 0 A_FireCustomMissile("PistolCaseSpawn",5,0,0,-9)
		PIST E 1
		PIST E 0 A_JumpIfInventory("Reloading",1,"Reload")
		PIST E 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
		PIST E 0 A_JumpIfInventory("Reloading",1,"Reload")
		PIST E 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
		PIST E 2 A_WeaponReady(WRF_NOBOB | WRF_ALLOWRELOAD)
		Goto Ready

	FireEndTactical:
		PIST E 0
		PIST E 0 A_JumpIfInventory("IsRunning", 1, "FireEndTactical2")
		PIST D 1
		PLAY A 0 ACS_NamedExecuteAlways("BDTacticalSpeed", 0, 0, 0, 0)//Makes player slower.
		PIST E 2 A_WeaponReady(1)
		PIST E 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
		Goto Ready

	FireEndTactical2:
		PIST E 0
		PLAY A 0 ACS_NamedExecuteAlways("BDSprint", 0, 0, 0, 0)//Makes player faster.
		PIST D 1
		PIST E 0 A_GiveInventory("UsedStamina", 1)
		PIST E 2 A_WeaponReady(1)
		PIST E 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
		Goto Sprinting

	NoAmmo:
		RIFG A 0
		RIFG A 0 A_ZoomFactor(1.0)
		RIFG A 0 A_Takeinventory("Zoomed",1)
		RIFG A 0 A_Takeinventory("ADSmode",1)
		RIFG A 0 A_JumpIfInventory("TurboReload",1,"TurboReload")
		TNT1 A 0 A_PlaySound("weapons/empty", 4)
		NoAmmo2:
		RIFG A 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
	    RIFG A 0 A_JumpIfInventory("Taunting",1,"Taunt")
		RIFG A 0 A_JumpIfInventory("Salute1", 1, "Salute")
		RIFG A 0 A_JumpIfInventory("Salute2", 1, "Salute")
		RIFG A 0 A_JumpIfInventory("ExecuteDownedEnemy", 1, "DoExecution")
		RIFG A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		PIST I 5 A_WeaponReady(WRF_ALLOWRELOAD)
		RIFG A 0 A_JumpIfInventory("Reloading",1,"Reload")
		RIFG A 0 A_JumpIfInventory("TurboReload",1,"TurboReload")
		TNT1 A 0 A_JumpIfInventory("NoAutoReload", 1, 2)
		RIFG A 0 A_JumpIfInventory("Clip1",1,"Reload")
		RIFG A 0 A_JumpIfInventory("IsRunning",1,"StartSprint")
		Goto Ready3

    Reload://round in chamber
		"####" "#" 1 A_WeaponReady
		PIST E 0 A_Takeinventory("Reloading",1)
		PIST E 0 A_Takeinventory("ADSmode",1)
		PIST E 0 A_Takeinventory("Zoomed",1)
        PIST E 0 A_ZoomFactor(1.0)
		PIST E 0 A_JumpIfInventory("BDPistolAmmo",16,"Ready")

        PIST E 0 A_JumpIfInventory("Clip1",1,1)
        Goto Ready
        TNT1 AAA 0
		PIST E 0 A_JumpIfInventory("TurboReload",1,"TurboReload")
		PIST E 0 A_JumpIfInventory("PistolUnloaded", 1, "PutMag2")
		PIST E 0 A_JumpIfInventory("BDPistolAmmo",1,1)
		Goto Reload2//reload empty chamber
		TNT1 AAA 0
		PIST E 0 A_PlaySound("Pistol/Reload1")
        PISR ABCDEF 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		PIST E 0 A_JumpIfInventory("BDPistolAmmo",2,"PutMag")//if mag is not empty then retain it
		PISR G 4 A_FireCustomMissile("EmptyPistolClipSpawn",0,0,0,-20)
	PutMag:
		PISR G 4
		PISR HIJKL 1
		PISR A 0 A_PlaySound("Pistol/Reload2")
		PISR MNOPQRSTUVW 1
		PISR DCBA 1
		PIST E 0 A_Takeinventory("PistolUnloaded",1)
		Goto InsertBullets

	TurboReload:
		PIST E 0 A_Takeinventory("PistolUnloaded",1)
        Goto InsertBullets//15+1

	InsertBullets://15+1
		TNT1 AAAA 0
		PIST E 0 A_JumpIfInventory("BDPistolAmmo",16,15)
		PIST E 0 A_JumpIfInventory("Clip1",1,3)
		Goto Ready
        TNT1 AAAAAA 0
		PIST E 0 A_Giveinventory("BDPistolAmmo",1)
		PIST E 0 A_Giveinventory("BDDualPistolAmmo",1)
		PIST E 0 A_Takeinventory("Clip1",1)
		Goto InsertBullets

		TNT1 AAAAAAAAAA 0
		PIST E 1 A_Takeinventory("Reloading",1)

		PIST E 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
        Goto Ready
		TNT1 AAAA 0
		PIST E 1 A_Takeinventory("Reloading",1)

		PIST E 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
        Goto Ready

	Reload2://Reload empty chamber
		PIST E 0 A_PlaySound("Pistol/Reload1")
        2ISR ABCDEF 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		2ISR G 4 A_FireCustomMissile("EmptyPistolClipSpawn",0,0,0,-20)
	PutMag2:
		2ISR G 4
		2ISR HIJKL 1
		2ISR A 0 A_PlaySound("Pistol/Reload2")
		2ISR MNOPQRS 1
		2ISR A 0 A_PlaySound("pistol/slide2")
		2ISR TUV 1
		PISR WDCBA 1
		PIST E 0 A_Takeinventory("PistolUnloaded",1)
		Goto InsertBullets2

	InsertBullets2://15
		TNT1 AAAA 0
		PIST E 0 A_JumpIfInventory("BDPistolAmmo",15,15)
		PIST E 0 A_JumpIfInventory("Clip1",1,3)
		Goto Ready
        TNT1 AAAAAA 0
		PIST E 0 A_Giveinventory("BDPistolAmmo",1)
		PIST E 0 A_Giveinventory("BDDualPistolAmmo",1)
		PIST E 0 A_Takeinventory("Clip1",1)
		Goto InsertBullets2

		TNT1 AAAAAAAAAA 0
		PIST E 1 A_Takeinventory("Reloading",1)
		PIST E 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
        Goto Ready
		TNT1 AAAA 0
		PIST E 1 A_Takeinventory("Reloading",1)

		PIST E 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
        Goto Ready

 	Spawn:
		PIST G -1
		Stop

	AltFire:
		TNT1 A 0
		TNT1 A 0 A_JumpIFInventory("PistolBurstFire", 1, "StopAlt")
		TNT1 A 0 A_GiveInventory("PistolBurstFire", 1)
		TNT1 A 0 A_Print("Burst fire")
		TNT1 A 0 A_PlaySound("weapons/empty", 4)
		Goto Ready3

	StopAlt:
		TNT1 A 0
		TNT1 A 0 A_TakeInventory("PistolBurstFire", 1)
		TNT1 A 0 A_Print("Single shot")
		TNT1 A 0 A_PlaySound("weapons/empty", 4)
		Goto Ready3

	Steady:
		TNT1 A 1
		Goto Ready

	Unload:
		"####" "#" 1 A_WeaponReady
		PIST E 0 A_Takeinventory("Unloading",1)
		PIST E 0 A_Takeinventory("ADSmode",1)
		PIST E 0 A_JumpIfInventory("BDPistolAmmo",1,3)
        Goto Ready
        TNT1 AAA 0
		PIST E 0 A_Takeinventory("Zoomed",1)
        PIST E 0 A_ZoomFactor(1.0)
		SMGG A 0 A_Giveinventory("PistolUnloaded",1)
		PIST E 0 A_Takeinventory("Reloading",1)
        PISR ABCDEWVUTS 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		PISR A 0 A_PlaySound("Pistol/Reload1")
		2ISR ONMLKJIHG 1 A_WeaponReady(WRF_NOBOB | WRF_NOFIRE)
		A12R A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		2ISR FEDCBA 1

	RemoveBullets:
		TNT1 AAAA 0
		PIST E 0 A_JumpIfInventory("BDPistolAmmo",1,3)
		Goto Ready
        TNT1 AAAAAA 0
		PIST E 0 A_Takeinventory("BDPistolAmmo",1)
		PIST E 0 A_Takeinventory("BDDualPistolAmmo",1)
		PIST E 0 A_Giveinventory("Clip1",1)
		Goto RemoveBullets

		TNT1 AAAAAAAAAA 0
		PIST E 0 A_Takeinventory("Reloading",1)

		PIST E 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
        Goto Ready
		TNT1 AAAA 0
		PIST E 0 A_Takeinventory("Reloading",1)

		PIST E 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
        Goto Ready

	DualWield:
		PIST E 1
		PIST E 0 A_Takeinventory("StartDualWield",1)
		//PIST E 0 A_JumpIfInventory("IsTacticalClass", 1, "NoDual")//Tactical mode cant dual
		PIST E 0 A_JumpIfInventory("HasDualPistol", 1, "GetDual")
		PIST E 0 A_Print("You must have two identical weapons to dual wield.")
		Goto Ready

	 GetDual:
		PIST E 0
		PIST E 0 A_Takeinventory("Zoomed",1)
		PIST E 0 A_Takeinventory("ADSmode",1)
		PIST E 0 A_TakeInventory("TossGrenade", 1)
		PIST E 0 A_TakeInventory("Kicking",1)
        PIST E 0 A_TakeInventory("Taunting",1)
        PIST E 0 A_TakeInventory("Reloading",1)
		PIST E 0 A_TakeInventory("Unloading",1)
		PIST E 0 A_TakeInventory("Salute1", 1)
		PIST E 0 A_TakeInventory("Salute2", 1)
		PIST E 0 A_GiveInventory("DualPistols",1)
        PIST E 0 A_ZoomFactor(1.0)
		PIST E 0 A_SelectWEapon("DualPistols")
		Goto Ready

	NoDual:
		PIST E 1
		PIST E 0 A_Print("You can't dual wield pistols when playing in Tactical Mode.",1)
		Goto Ready

	KickFlash://18 tics
		PISS DEF 1
		PISS F 12
		PISS FED 1
		Stop

	SlideFlashEnd://10 tics
		PISS F 7
		PISS FED 1
		Stop
	}
}

ACTOR BDPistolAmmo : Ammo
{
   Inventory.Amount 0
   Inventory.MaxAmount 16
   Ammo.BackpackAmount 0
   Ammo.BackpackMaxAmount 16
   Inventory.Icon PISTG0
}
