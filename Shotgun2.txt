ACTOR Shot_Gun2 : BrutalWeapon
{
	Game Doom
	Weapon.SelectionOrder 1300
	Weapon.AmmoUse1 0
	Weapon.AmmoUse2 0
	Weapon.AmmoGive1 0
	Weapon.AmmoGive2 0
	Weapon.AmmoType1 "AmmoSlug"
	Weapon.AmmoType2 "ShotgunAmmo"
	Inventory.PickupMessage "$GOTSHOTGUN"
    Inventory.PickupSound "weapons/sgpump"
	Obituary "$OB_MPSHOTGUN"
    AttackSound "None"
    +WEAPON.NOAUTOAIM
	+WEAPON.NOAUTOFIRE
	+WEAPON.NOALERT
	+FORCEXYBILLBOARD
	Scale 0.8
	Tag "Shotgun"
	DropItem "ShotgunSpawner"
	States
	{
	KickFlash://18 tics
		SHTA NOP 1
		SHTA Q 12
		SHTA PON 1
		Stop

	SlideFlashEnd://10 tics
		SHTA Q 7
		SHTA PON 1
		Stop

	Steady:
		TNT1 A 1
		Goto Ready

    Spawn:
		SHTC A -1
		Stop

	Select:
		SHTN A 0 A_Giveinventory("GoSpecial",1)
		TNT1 A 0 A_TakeInventory("FistsSelected", 1)
		TNT1 A 0 A_GiveInventory("ShotgunSelected", 1)
		TNT1 A 0 A_TakeInventory("MinigunSelected", 1)
		TNT1 A 0 A_TakeInventory("SSGSelected", 1)
		TNT1 A 0 A_TakeInventory("RocketLauncherSelected", 1)
		TNT1 A 0 A_TakeInventory("GrenadeLauncherSelected", 1)
		TNT1 A 0 A_TakeInventory("PlasmaGunSelected", 1)
		TNT1 A 0 A_TakeInventory("RailGunSelected", 1)
		TNT1 A 0 A_TakeInventory("RevenantLauncherSelected", 1)
		TNT1 A 0 A_TakeInventory("SubMachineGunSelected", 1)
		TNT1 A 0 A_TakeInventory("BFG10KSelected", 1)
		TNT1 A 0 A_TakeInventory("BFGSelected", 1)
		TNT1 A 0 A_TakeInventory("FlameCannonSelected", 1)
		TNT1 A 0 A_TakeInventory("SawSelected", 1)
		SHTN A 0 A_Takeinventory("HasBarrel",1)
		SHTN A 0 A_TakeInventory("TossGrenade", 1)
		SHTN A 0 A_TakeInventory("Shot_Gun", 1)
		A12G A 0 A_GunFlash
		TNT1 AAAAAAAAA 0 A_Raise
		TNT1 AAAA 1 A_Raise

	SelectAnimation:
        TNT1 A 0
		TNT1 A 0 A_JumpIfInventory("GoFatality", 1, "Steady")
        SHTA MLKJ 1 A_JumpIfInventory("ShotgunChangeAmmo",1,"Reload")
		SHTN A 0 A_PlaySound("weapons/sgpump")

	Ready3:
	Ready:
		TNT1 A 0
		SHTN A 0 A_JumpIfInventory("Zoomed",1,"Ready2")
		SHTN A 1 A_WeaponReady(WRF_ALLOWRELOAD)
        SHTN A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        SHTN A 0 A_JumpIfInventory("Taunting",1,"Taunt")
        SHTN A 0 A_JumpIfInventory("Reloading",1,"Reload")
		SHTN A 0 A_JumpIfInventory("Salute1", 1, "Salute")
		SHTN A 0 A_JumpIfInventory("Salute2", 1, "Salute")
		SHTN A 0 A_JumpIfInventory("Unloading",1,"Unload")
		SHTN A 0 A_JumpIfInventory("StartDualWield",1,"ChangeShell")
		SHTN A 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
	Ready3:
		SHTN A 1 A_WeaponReady(WRF_ALLOWRELOAD)
		SHTN A 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
		TNT1 A 0 A_JumpIfInventory("ExecuteDownedEnemy", 1, "DoExecution")
		SHTN A 0 A_JumpIfInventory("IsRunning",1,"StartSprint")
        Goto Ready

	Ready2:
		TNT1 A 0
		SHT8 C 1 A_WeaponReady(WRF_ALLOWRELOAD)
        SHTN A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        SHTN A 0 A_JumpIfInventory("Taunting",1,"Taunt")
        SHTN A 0 A_JumpIfInventory("Reloading",1,"Reload")
		SHTN A 0 A_JumpIfInventory("Salute1", 1, "Salute")
		SHTN A 0 A_JumpIfInventory("Salute2", 1, "Salute")
		SHTN A 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
		TNT1 A 0 A_JumpIfInventory("ExecuteDownedEnemy", 1, "DoExecution")
		SHTN A 0 A_JumpIfInventory("Unloading",1,"Unload")
		SHTN A 0 A_JumpIfInventory("StartDualWield",1,"ChangeShell")
		SHTN A 0 A_JumpIfInventory("IsRunning",1,"StartSprint")
		SHTN A 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
		Loop

	Deselect:
		SHTN A 0
		SHTN A 0 A_Takeinventory("Zoomed",1)
		SHTN A 0 A_TakeInventory("TossGrenade", 1)
		SHTN A 0 A_Takeinventory("ADSmode",1)
		SHTN A 0 A_Takeinventory("UseShotgunStrap",1)
        SHTN A 0 A_ZoomFactor(1.0)
		RIFG A 0 A_SetCrosshair(0)
		SHTA NOPQ 1
		TNT1 AAAAAAAAAAAAAAAAAA 0 A_Lower
		TNT1 A 1 A_Lower
		Wait

    Fire:
        SHTN A 0 A_Takeinventory("Reloading",1)
		SHTN A 0 A_JumpIfInventory("Zoomed",1,"Fire2")
        SHTN A 0 A_JumpIfInventory("ShotgunAmmo",1,1)
        Goto NoAmmo
        TNT1 AAA 0
		SHTN A 0 A_JumpIfInventory("ShotgunWasEmpty", 1, "EjectEmpty")
        SHTN A 0 A_AlertMonsters
        SHTN A 0 A_PlaySound("SlugShot", 1)
		RIFF A 0 A_FireCustomMissile("DistantFireSoundShotgun", random(-1,1), 0, 0, -12, 0, random(-1,1))
        SHTN A 0
		SHTN A 0 A_SpawnItemEx("PlayerMuzzle1",32,0,28)
		TNT1 AAAAAAAAAAAAAAAA 0 A_FireCustomMissile("ShotgunParticles", random(-12,12), 0, -1, 0, 0, random(-9,9))
		TNT1 AAAAA 0 A_FireCustomMissile("ShotgunParticles2", random(-12,12), 0, -1, 0, 0, random(-9,9))
		TNT1 AAAAA 0 A_FireCustomMissile("DecorativeTracer", random(-3,3), 0, 0, 0, 0, random(-3,3))
		SHTN A 0 A_Takeinventory("ShotgunAmmo",1)
		SHTA A 1 BRIGHT
		SHTN A 0 A_FireBullets (0,0,1,40,"shotpuff",FBF_NORANDOM,128)
		SHTN A 0 A_FireBullets (1.3, 1.4, -1, 120, "ShotgunPuff", FBF_NORANDOM)
		SHTN A 0 A_JumpIfInventory("IsNOTTacticalClass", 1, 2)//Tactical Mode deals extra recoil
		SHTN A 0 A_SetPitch(pitch-1.8)
		SHTN A 0 A_SetPitch(pitch-1.8)
		TNT1 A 0 A_ZoomFactor(0.98)
		SHTA B 1 BRIGHT
		TNT1 A 0 A_ZoomFactor(1.0)
		SHTA C 1 A_SetPitch(0.6 + pitch)
		SHTA DE 1 A_SetPitch(0.6 + pitch)
		SHTN A 1
		Goto Eject

   Eject:
		SHTN A 0
		SHTN A 0 A_GiveInventory("Pumping", 1)
		SHTN A 0 A_GiveInventory("GoSpecial", 1)
		SHTN BCDEFG 1 A_WeaponReady(WRF_NOFIRE| WRF_NOBOB)//Allows quick switch
		SHTN A 0 A_JumpIfInventory("Reloading",1,"InsertingShells")
		SHTN A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHTN A 0 A_PlaySound("weapons/sgpump", 3)
		SHTN R 1 A_WeaponReady(WRF_NOFIRE| WRF_NOBOB)//Allows quick switch
		SHTA YX 1 A_WeaponReady(WRF_NOFIRE| WRF_NOBOB)//Allows quick switch
		SHTN J 1 A_WeaponReady(WRF_NOFIRE| WRF_NOBOB)//Allows quick switch
		SHTN A 0 A_FireCustomMissile("SlugCaseSpawn",40,0,14,0)
		SHTN A 0 A_JumpIfInventory("Reloading",1,"InsertingShells")
		SHTN A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHTN A 0 A_JumpIfInventory("ShotgunAmmo",1,"Chamber")
		Goto ChamberEmpty

   EjectEmpty:
		SHTN A 0
		SHTN A 0 A_Takeinventory("Zoomed",1)
		SHTN A 0 A_ZoomFactor(1.0)
		RIFG A 0 A_SetCrosshair(0)
		SHTN A 0 A_GiveInventory("Pumping", 1)
		SHTN A 0 A_GiveInventory("GoSpecial", 1)
		SHTN BCDEFG 1 A_WeaponReady(WRF_NOFIRE| WRF_NOBOB)//Allows quick switch
		SHTN A 0 A_JumpIfInventory("Reloading",1,"InsertingShells")
		SHTN A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHTN A 0 A_PlaySound("weapons/sgpump", 3)
		SHTN HIJ 1 A_WeaponReady(WRF_NOFIRE| WRF_NOBOB)//Allows quick switch
		SHTN A 0 A_JumpIfInventory("Reloading",1,"InsertingShells")
		SHTN A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHTN A 0 A_JumpIfInventory("ShotgunAmmo",1,"Chamber")
		Goto ChamberEmpty

	Chamber:
		TNT1 A 0
		SHTN A 0 A_TakeInventory("ShotgunWasEmpty",1)
		SHTN KLMNO 1 A_WeaponReady(WRF_NOFIRE| WRF_NOBOB)//Allows quick switch
		SHTA XY 1 A_WeaponReady(WRF_NOFIRE| WRF_NOBOB)//Allows quick switch
		SHTN R 1 A_WeaponReady(WRF_NOFIRE| WRF_NOBOB)//Allows quick switch
		SHTN GFEDCB 1 A_WeaponReady(WRF_NOFIRE| WRF_NOBOB)//Allows quick switch
		SHTN A 0 A_JumpIfInventory("Reloading",1,"InsertingShells")
		SHTN A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		Goto Ready3

   ChamberEmpty:
		SHTN A 0
		SHTN A 0 A_Takeinventory("Zoomed",1)
		SHTN A 0 A_ZoomFactor(1.0)
		RIFG A 0 A_SetCrosshair(0)
		SHTN A 0 A_Giveinventory("ShotgunWasEmpty",1)
		SHTN A 0 A_JumpIfInventory("TurboReload", 1, "TurboReload")
		SHTN A 0 A_JumpIfInventory("NoAutoReload", 1, 1)
		Goto InsertingShells
		TNT1 AAA 0
		SHTN KLMNOIH 1 A_WeaponReady(WRF_NOFIRE| WRF_NOBOB)//Allows quick switch
		SHTN GFEDCB 1 A_WeaponReady(WRF_NOFIRE| WRF_NOBOB)//Allows quick switch
		SHTN A 0 A_JumpIfInventory("Reloading",1,"InsertingShells")
		SHTN A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		Goto NoAmmo

	Fire2:
		SHTN A 0
        SHTN A 0 A_JumpIfInventory("ShotgunAmmo",1,1)
        Goto NoAmmo3
        TNT1 AAA 0
		SHTN A 0 A_JumpIfInventory("ShotgunWasEmpty", 1, "EjectEmpty")
        SHTN A 0 A_AlertMonsters
        SHTN A 0 A_PlaySound("SlugShot", 1)
		RIFF A 0 A_FireCustomMissile("DistantFireSoundShotgun", random(-1,1), 0, 0, -12, 0, random(-1,1))
        SHTN A 0
		SHTN A 0 A_SpawnItemEx("PlayerMuzzle1",32,0,28)
		TNT1 AAAAAAAAAAAAAAAA 0 A_FireCustomMissile("ShotgunParticles", random(-12,12), 0, -1, 0, 0, random(-9,9))
		TNT1 AAAAA 0 A_FireCustomMissile("ShotgunParticles2", random(-12,12), 0, -1, 0, 0, random(-9,9))
		TNT1 AAAAA 0 A_FireCustomMissile("DecorativeTracer", random(-3,3), 0, 0, 0, 0, random(-3,3))
		SHTN A 0 A_Takeinventory("ShotgunAmmo",1)
		SHT8 D 1 BRIGHT
		SHTN A 0 A_FireBullets (0,0,1,40,"shotpuff",FBF_NORANDOM,128)
		SHTN A 0 A_FireBullets (0.8, 0.9, -1, 120, "ShotgunPuff", FBF_NORANDOM)
		SHTN A 0 A_JumpIfInventory("IsNOTTacticalClass", 1, 2)//Tactical Mode deals extra recoil
		SHTN A 0 A_SetPitch(pitch-0.9)
		SHTN A 0 A_SetPitch(pitch-0.9)
		TNT1 A 0 A_ZoomFactor(1.29)
		SHT8 E 1
		TNT1 A 0 A_ZoomFactor(1.3)
		SHT8 FGC 1 A_SetPitch(0.3 + pitch)
		SHT8 C 1
		Goto EjectADS

   EjectADS:
		SHTN A 0
		SHTN A 0 A_GiveInventory("Pumping", 1)
		SHTN A 0 A_GiveInventory("GoSpecial", 1)
		SHTN A 0 A_PlaySound("weapons/sgpump", 5)
		SHT8 GHIJ 2 A_WeaponReady(WRF_NOFIRE| WRF_NOBOB)//Allows quick switch
		SHTN A 0 A_FireCustomMissile("SlugCaseSpawn",40,0,14,0)
		SHTN A 0 A_JumpIfInventory("Reloading",1,"InsertingShells")
		SHTN A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		Goto ChamberADS

	ChamberADS:
		TNT1 AAA 0
		SHTN A 0 A_TakeInventory("ShotgunWasEmpty", 1)
		SHT8 JIHG 2 A_WeaponReady(WRF_NOFIRE| WRF_NOBOB)//Allows quick switch
		SHTN A 0 A_JumpIfInventory("Reloading",1,"InsertingShells")
		SHTN A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHTN A 0 A_JumpIfInventory("FiredSecondary",1,"HoldAim")//If still pressing alt fire, it means its holding aim
		SHTN A 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
		Goto Ready2

	AltFire:
        SHTN A 0
		SHTN A 0 A_Giveinventory("GoSpecial",1)
		SHTN A 0 A_JumpIfInventory("Zoomed",1,"NoAim")
		SHTN A 0 A_Giveinventory("Zoomed",1)
        SHTN A 0 A_ZoomFactor(1.3)
		SHTN A 0 A_Giveinventory("ADSmode",1)
		RIFG A 0 A_SetCrosshair(41)
        SHT8 AB 1
        SHTN A 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
		SHT8 C 1
		SHTN A 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
		SHT8 C 1
		SHTN A 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
		SHT8 C 1
		SHTN A 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
		SHT8 C 1
		SHTN A 0 A_JumpIfInventory("FiredSecondary",1,"HoldAim")
        Goto Ready2

	HoldAim:
	    SHTN A 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
		SHT8 C 1 A_WeaponReady(WRF_ALLOWRELOAD)
		SHTN A 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
		SHT8 C 1 A_WeaponReady(WRF_ALLOWRELOAD)
		SHTN A 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
		SHT8 C 1 A_WeaponReady(WRF_ALLOWRELOAD)
		SHTN A 0 A_JumpIfInventory("FiredPrimary",1,"Fire")
		SHT8 C 1 A_WeaponReady(WRF_ALLOWRELOAD)
		SHTN A 0 A_JumpIfInventory("FiredSecondary",1,"HoldAim")

	NoAim:
        TNT1 AAAAAA 0
		SHTN A 0 A_Takeinventory("Zoomed",1)
        SHTN A 0 A_ZoomFactor(1.0)
		SHTN A 0 A_Takeinventory("ADSmode",1)
		RIFG A 0 A_SetCrosshair(0)
		SHTN A 0 A_Giveinventory("GoSpecial",1)
		SHT8 BA 1
        Goto Ready

    ReloadWithNoAmmoLeft:
    Reload:
	    SHTN A 0
		SHTN A 0 A_TakeInventory("Reloading", 1)
		SHTN A 0 A_JumpIfInventory("AmmoSlug", 1, 1)
		Goto Ready
		SHTN AAA 0
	    SHTN A 0 A_Takeinventory("Zoomed",1)
        SHTN A 0 A_ZoomFactor(1.0)
		RIFG A 0 A_SetCrosshair(0)
		SHTN A 1 A_WeaponReady(WRF_NOBOB)
		SHTN A 0 A_JumpIfInventory("ShotgunAmmo",9,"Ready3")
		SHTN A 0 A_JumpIfInventory("ShotgunAmmo", 1, "ReloadNormally")//Check if there is a shell in the chamber
		SHTN A 0 A_GiveInventory("ShotgunWasEmpty", 1)//This means there is no shell in the chamber

	ReloadNormally:
        SHTN A 0 A_JumpIfInventory("AmmoSlug",1,1)//If there is no reserve ammo, don't reload, go back to Ready
        Goto Ready
		TNT1 AAA 0
		SHTN A 0 A_Takeinventory("ShotgunUnloaded",1)
		TNT1 A 0 A_TakeInventory("ShotgunChangeAmmo",1)
		SHTN A 0 A_JumpIfInventory("TurboReload", 1, "TurboReload")//Check if reloads are disabled.
		SHTN A 0 A_GiveInventory ("Pumping", 1)
		SHTN BCDEFGSTU 1 A_WeaponReady(WRF_NOBOB)
		SHTN A 0 A_JumpIfInventory("Kicking",1,"DoKick")

	 InsertingShells:
		SHTN A 0
		SHTN A 0 A_JumpIfInventory("AmmoSlug",1,1)//if there are no more reserve shells, finish reload.
        Goto FinishedInsertingShells
		SHTN A 0
		SHTN A 0 A_GiveInventory ("Pumping", 1)
		SHTN A 0 A_TakeInventory("Reloading", 1)
        SHTN A 0 A_JumpIfInventory("ShotgunAmmo",8,"CheckIfFinishReload")
        SHTN A 0 A_Giveinventory("ShotgunAmmo",1)
		SHTN A 0 A_Takeinventory("AmmoSlug",1)

	InsertShellAnimation:
		SHTN V 1 A_WeaponReady(WRF_NOBOB)
		SHTA Z 1 A_WeaponReady(WRF_NOBOB)
		SHTN A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHTN A 0 A_PlaySound("insertshell")
		SHTN XYZ 1 A_WeaponReady(WRF_NOBOB)
		SHTN "[]" 1 A_WeaponReady(WRF_NOBOB)
		SHTN A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHTN A 0 A_JumpIfInventory("ShotgunAmmo",9,"FinishedInsertingShells")//If there are 8+1 shells, stop it immediately
        Goto InsertingShells

	CheckIfFinishReload:
		SHTN A 0
		SHTN A 0 A_JumpIfInventory("ShotgunWasEmpty", 1, "EjectEmpty")//If shotgun was empty,
		//Play the insert shell animation again, and insert the 9th shell
		SHTN A 0 A_JumpIfInventory("AmmoSlug",1,1)//if there are no more reserve shells, finish reload.
        Goto FinishedInsertingShells
		SHTN A 0 A_Giveinventory("ShotgunAmmo",1)
		SHTN A 0 A_Takeinventory("AmmoSlug",1)
        Goto InsertShellAnimation

	FinishedInsertingShells:
		SHTN EDCBA 1 A_WeaponReady(WRF_NOBOB)
		SHTN A 0 A_Takeinventory("Reloading",1)
		Goto Ready

	TurboReload:
	TurboInsertShells:
		SHTN A 0 A_JumpIfInventory("AmmoSlug",1,1)//if there are no more reserve shells, finish reload.
        Goto FinishTurboReload
		SHTN AAA 0
		SHTN A 0 A_GiveInventory ("Pumping", 1)
		SHTN A 0 A_TakeInventory("Reloading", 1)
        SHTN A 0 A_JumpIfInventory("ShotgunAmmo",9,"FinishTurboReload")
        SHTN A 0 A_Giveinventory("ShotgunAmmo",1)
		SHTN A 0 A_Takeinventory("AmmoSlug",1)
		Goto TurboInsertShells
	FinishTurboReload:
		SHTN A 0 A_Takeinventory("Reloading",1)
		SHTN A 1 A_JumpIfInventory("ShotgunWasEmpty", 1, "EjectEmpty")
		Goto Ready

	ChangeShell:
		SHTN A 1
		TNT1 A 0 A_TakeInventory("StartDualWield", 1)
		TNT1 A 0 A_GiveInventory("ShotgunChangeAmmo", 1)
        TNT1 A 0 A_ZoomFactor(1.0)
		TNT1 A 0 A_Takeinventory("Zoomed",1)
		TNT1 A 0 A_Takeinventory("ADSmode",1)
		RIFG A 0 A_SetCrosshair(0)
		TNT1 A 0 A_JumpIfInventory("AmmoShell", 1, "Unload")
		Goto NoShells

	NoShells:
		TNT1 A 0 A_Print("You are out of shotgun shells.", 2)
		SHTN A 0 A_TakeInventory("ShotgunChangeAmmo",1)
		SHTN A 3
		Goto Ready

	Unload:
		SHTN A 1
		SHTN A 0 A_ZoomFactor(1.0)
		SHTN A 0 A_Takeinventory("Unloading",1)
		SHTN A 0 A_Takeinventory("ADSmode",1)
		SHTN A 0 A_Takeinventory("Zoomed",1)
		RIFG A 0 A_SetCrosshair(0)
        SHTN A 0 A_JumpIfInventory("ShotgunAmmo",1,2)
		TNT1 A 0 A_JumpIfInventory("ShotgunChangeAmmo", 1, "FinishUnload")
        Goto Ready
        TNT1 AAA 0
		SHTN A 0 A_Takeinventory("Zoomed",1)
		SHTN A 0 A_GiveInventory ("Pumping", 1)
		SHTN A 0 A_Takeinventory("Unloading",1)
		SHTN BCDEFG 1 A_WeaponReady(WRF_NOBOB)
		SHTN A 0 A_JumpIfInventory("Kicking",1,"DoKick")

	RemoveBullets:
		TNT1 A 0
        TNT1 AAA 0
		SHTN A 0 A_Takeinventory("ShotgunAmmo",1)
		SHTN A 0 A_Giveinventory("AmmoSlug",1)
		SHTN A 0 A_PlaySound("weapons/sgpump", 3)
		SHTN R 1 A_WeaponReady(WRF_NOBOB)
		SHTA YX 1 A_WeaponReady(WRF_NOBOB)
		SHTN O 1 A_WeaponReady(WRF_NOBOB)
		SHTN A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHTN A 0 A_JumpIfInventory("ShotgunAmmo",1,1)
		Goto FinishUnload
		SHTN O 2 A_WeaponReady(WRF_NOBOB)
		SHTA XY 2 A_WeaponReady(WRF_NOBOB)
		SHTN RG 2 A_WeaponReady(WRF_NOBOB)
		Goto RemoveBullets

	FInishUnload:
		SHTN JIHG 2 A_WeaponReady(WRF_NOBOB)
		SHTN FEDCBA 1 A_WeaponReady(WRF_NOBOB)
		SHTN A 0 A_Takeinventory("Unloading",1)
		SHTN A 0 A_GiveInventory("ShotgunUnloaded",1)
		SHTN A 0 A_Giveinventory("ShotgunWasEmpty",1)
		TNT1 A 0 A_JumpIfInventory("ShotgunChangeAmmo", 1, "ChangeAmmo")
		Goto Ready3

	ChangeAmmo:
		TNT1 A 0
		TNT1 A 0 A_GiveInventory("Shot_Gun")
		TNT1 A 0 A_GiveInventory("Reloading",1)
		TNT1 A 0 A_SelectWeapon("Shot_Gun")
		Goto Ready3

	NoAmmo:
		RIFG A 0
		RIFG A 0 A_ZoomFactor(1.0)
		RIFG A 0 A_Takeinventory("Zoomed",1)
		RIFG A 0 A_Takeinventory("ADSmode",1)
		RIFG A 0 A_SetCrosshair(0)
		RIFG A 0 A_JumpIfInventory("TurboReload",1,"TurboReload")
		TNT1 A 0 A_PlaySound("weapons/empty", 4)
		NoAmmo2:
		RIFG A 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
	    RIFG A 0 A_JumpIfInventory("Taunting",1,"Taunt")
		RIFG A 0 A_JumpIfInventory("Salute1", 1, "Salute")
		RIFG A 0 A_JumpIfInventory("Salute2", 1, "Salute")
		RIFG A 0 A_JumpIfInventory("ExecuteDownedEnemy", 1, "DoExecution")
		RIFG A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHTN A 5 A_WeaponReady(WRF_ALLOWRELOAD | WRF_NOFIRE)
		RIFG A 0 A_JumpIfInventory("Reloading",1,"Reload")
		RIFG A 0 A_JumpIfInventory("TurboReload",1,"TurboReload")
		TNT1 A 0 A_JumpIfInventory("NoAutoReload", 1, 2)
		RIFG A 0 A_JumpIfInventory("AmmoSlug",1,"Reload")
		RIFG A 0 A_JumpIfInventory("IsRunning",1,"StartSprint")
		Goto Ready3

	NoAmmo3:
		RIFG A 0 A_JumpIfInventory("TossGrenade",1,"TossGrenade")
	    RIFG A 0 A_JumpIfInventory("Taunting",1,"Taunt")
		RIFG A 0 A_JumpIfInventory("Salute1", 1, "Salute")
		RIFG A 0 A_JumpIfInventory("Salute2", 1, "Salute")
		RIFG A 0 A_JumpIfInventory("ExecuteDownedEnemy", 1, "DoExecution")
		RIFG A 0 A_JumpIfInventory("Kicking",1,"DoKick")
		SHT8 A 5 A_WeaponReady(WRF_ALLOWRELOAD | WRF_NOPRIMARY)
		RIFG A 0 A_JumpIfInventory("Reloading",1,"Reload")
		RIFG A 0 A_JumpIfInventory("TurboReload",1,"TurboReload")
		TNT1 A 0 A_JumpIfInventory("NoAutoReload", 1, 2)
		RIFG A 0 A_JumpIfInventory("AmmoSlug",1,"Reload")
		RIFG A 0 A_JumpIfInventory("IsRunning",1,"StartSprint")
		Goto Ready2

	StartSprint:
		SHTA RST 1 A_WeaponReady(WRF_ALLOWRELOAD)
		SHTN A 0 A_Takeinventory("Zoomed",1)
        SHTN A 0 A_ZoomFactor(1.0)
		SHTN A 0 A_Takeinventory("ADSmode",1)
		RIFG A 0 A_SetCrosshair(0)
		SHTN A 0 A_Giveinventory("GoSpecial",1)
		TNT1 A 0 ACS_NamedExecuteAlways("BDSprint", 0, 0, 0, 0)//Makes player faster.

	Sprinting:
		SHTA U 1 offset(-9,34) A_SetPitch(pitch -0.5)
		SHTA U 1 offset(-6,36) A_SetPitch(pitch -0.5)
		SHTA U 1 offset(-3,38) A_SetPitch(pitch -0.5)
		SHTA U 1 offset(0,38) A_SetPitch(pitch -0.5)
		SHTA U 1 offset(3,36) A_SetPitch(pitch -0.5)
		SHTA U 1 offset(6,34) A_SetPitch(pitch -0.5)
		SHTA U 1 offset(9,32) A_SetPitch(pitch -0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 2)
		SMGG A 0 A_WeaponReady(WRF_NOBOB)
		SMGG A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        SMGG A 0 A_JumpIfInventory("Taunting",1,"Taunt")
        SMGG A 0 A_JumpIfInventory("Reloading",1,"Reload")

		SHTA U 1 offset(9,34) A_SetPitch(pitch +0.5)
		SHTA U 1 offset(6,36) A_SetPitch(pitch +0.5)
		SHTA U 1 offset(3,38) A_SetPitch(pitch +0.5)
		SHTA U 1 offset(0,38) A_SetPitch(pitch +0.5)
		SHTA U 1 offset(-3,36) A_SetPitch(pitch +0.5)
		SHTA U 1 offset(-6,34) A_SetPitch(pitch +0.5)
		SHTA U 1 offset(-9,32) A_SetPitch(pitch +0.5)
		TNT1 A 0 A_GiveInventory("UsedStamina", 2)
		SMGG A 0 A_WeaponReady(WRF_NOBOB)
		SMGG A 0 A_JumpIfInventory("Kicking",1,"DoKick")
        SMGG A 0 A_JumpIfInventory("Taunting",1,"Taunt")
        SMGG A 0 A_JumpIfInventory("Reloading",1,"Reload")

		PLAY A 0 ACS_NamedExecuteAlways("BDSprint", 0, 0, 0, 0)//Makes player faster.
		SMGG A 0 A_JumpIfInventory("IsRunning", 1, "Sprinting")
		Goto StopSprint

	StopSprint:
		SHTA A 0 A_JumpIfInventory("PowerStrength", 1, 2)
		SHTN A 0 A_JumpIfInventory("UsedStamina", 100, "StopSprintTired")
		SHTA TSR 1 ACS_NamedExecuteAlways("BDTacticalSpeed", 0, 0, 0, 0)//Makes player slower.
		Goto Ready

	StopSprintTired:
		SHTA TSR 1 ACS_NamedExecuteAlways("BDTacticalSpeed", 0, 0, 0, 0)//Makes player slower.
		SHTN A 0 A_PlaySound("Tired", 2)
		Goto Ready
	}
